export const GET_TRACKING_STATUSBAR = status => {

export const ARE_EVENTS_SAME = (event, currentEvent) => {
	const { actual_date, planned_date, event_type } = event || {};
	const {
		actual_date: currentActualData,
		planned_date: currentPlannedDate,
		event_type: currentEventType,
	} = currentEvent || {};
	return (
		actual_date === currentActualData
		&& planned_date === currentPlannedDate
		&& event_type === currentEventType
	);
};

// if completed => return actual date of 4.0 else return ETA
export const GET_ARRIVED_OR_ETA_DATE = (trackingStatus, shiploadEta, events, isRoad) => {
	if (IS_TRACKING_COMPLETED(trackingStatus)) {
		const date = isRoad
			? (events[CONST_EVENTS.goodsUnloaded] || {}).actual_date
			: (events[CONST_EVENTS.arrival] || {}).actual_date;
 
		if (date) {
			return (
				<Tooltip title={FORMAT_DATE_VIEW_TRACKING(date)}>
					<Ellipsis>{FORMAT_DATE_VIEW_TRACKING(date)}</Ellipsis>
				</Tooltip>
			);
		}
		return '--';
	}
 
	return FORMAT_DATE_VIEW_TRACKING(shiploadEta);
};
