	async componentDidMount() {
		const { getTrackingDetailsFn, queryParam } = this.props;
		const { id } = queryParam || {};
		const params = new URLSearchParams(window.location.search);
		const matchQueryObject = window.matchMedia(`(max-width: ${breakpoint.MD}px)`);
		this.setState({
			isMobileView: matchQueryObject.matches,
		});
 
		getTrackingDetailsFn(id, this.userType);
		this.scrollToCurrentTimeline();
		this.getFeatureSnippetsData();
 
		if (params.has('shared')) {
			analyticsTrack('Tracking Details Sharable Link Used', {});
		}
		if (params.has('whatsapp_redirect')) {
			analyticsTrack('Tools tracking details whatsapp redirect', {});
		}
 
		setTimeout(() => {
			this.setState({ isSurveyVisible: true });
		}, 30 * 1000);
 
		setTimeout(() => {
			this.shouldShowFeedbackInterviewModal();
		}, 15 * 1000);
	}
 
	componentDidUpdate(prevProps = {}, prevState = {}) {
		const { refreshTimer, isRefreshing, portCongestionData } = this.state;
		const { portCongestionData: prevPortCongestionData } = prevState;
		const {
			isCourierTrackingMapEnabled,
			requestStatus,
			containersDetails = {},
			mapData = {},
		} = this.props;
 
		const { containersDetails: prevContainerDetails, mapData: prevMapData } = prevProps || {};
 
		if (!isEqual(prevContainerDetails, containersDetails)) {
			if (isRefreshing) {
				/* stopped polling for updated details */
				clearInterval(refreshTimer);
 
				// eslint-disable-next-line react/no-did-update-set-state
				this.setState({ isRefreshing: false });
			}
 
			this.scrollToCurrentTimeline();
		}
		if (
			(containersDetails && !isEqual(prevContainerDetails?.pod, containersDetails?.pod))
			|| !isEqual(portCongestionData, prevPortCongestionData)
		) {
			this.calculatePortDelay(containersDetails?.pod);
		}
 
		if ((requestStatus || {}).getTrackingDetailsMapdata === 'SUCCESS' && prevMapData !== mapData) {
			this.getActiveKey();
		} else if (
			!isEqual(prevContainerDetails, containersDetails)
			&& !isCourierTrackingMapEnabled
			&& containersDetails.mode === 'courier'
		) {
			this.getActiveKey(true);
		}
	}
 
	getFeatureSnippetsData = async () => {
		try {
			const [, combinedPosts] = (await getWordpressBlogs()) || [];
			let portCongestionData = [];
			const portCongestionRes = await getRequest(trackingRootUrl, 'public/port-congestion-stats');
			portCongestionData = portCongestionRes?.data || [];
			portCongestionData.forEach((data, i) => {
				const { weekly_median_delay, fortnightly_median_delay, monthly_median_delay } = data;
				const delay = Math.round(
					weekly_median_delay || fortnightly_median_delay || monthly_median_delay || 0,
				);
				const delayToDisplay = delay + (delay > 1 ? ' days' : ' day');
				portCongestionData[i].port = {
					...portCongestionData[i].port,
					delay,
					delayToDisplay,
				};
			});
			this.setState({
				combinedPosts,
				portCongestionData,
			});
		} catch (error) {
			console.log('Error in port congestion api', error);
		}
	};

	};

			if (currentTimeline) {
				currentTimeline.parentElement.scrollTop = currentTimeline.offsetTop - 100;
			}
		}
	};

		const { shiploads } = containersDetails || {};
		const shiploadIndex = (shiploads || []).findIndex(({ id }) => id === currentShipId);
 
		this.setState({
			currentShiploadIndex: shiploadIndex,
			currentShiploadId: currentShipId,
			currentVesselId: currentVesId,
		});
	};

			const width = window.innerWidth > 0 ? window.innerWidth : window.screen.width;
			if (width < breakpoint.SM) {
				this.setState({ activeKey: 'tracking-milestone' });
				return;
			}
		}
 
		if (hideMapView) {
			this.setState({ activeKey: 'tracking-table-view' });
		} else {
			this.setState({
				activeKey: shouldShowMapView ? 'tracking-map-view' : 'tracking-table-view',
			});
		}
	};
