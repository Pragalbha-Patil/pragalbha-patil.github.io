import PropTypes from 'prop-types';
import { Row, Tooltip } from 'antd';
import { Archive, Unarchive } from 'common-util/icons/archive-unarchive';
import { StatusBar } from 'common-util/status-bar';
import { LocationAddress } from 'common-util/reusable-components';
import EnquiryTooltip from 'common-util/product/enquiry/components/EnquiryTooltip';
import { TooltipConfirmedFF } from 'common-util/product/enquiry/components';
import {
	GET_STATUS_TYPE,
	isEnquiryCancelled,
	getListBiddingTime,
	isSurfaceEnquiry,
	isDomesticSurfaceEnquiry,
	isEnquiryCopyEnabled,
} from 'common-util/product/enquiry/functions';
import EnquiriesLoader from 'common-util/shimmer/Enquiries';
import {
	EnquiryCard,
	RowTwo,
	CardColumn,
	QuoteName,
	CreatedBy,
	ShipmentInfo,
	EditIcons,
	FreightNotRequired,
	BidCloseTime,
	VendorQuotes,
	NoData,
	CenteredText,
	CardColumnFlexStart,
} from 'common-util/reusable-styles';
import EnquiryApproversToolTip from 'common-util/product/enquiry/components/EnquiryApproversToolTip';
import { StatusCol, NewRFQStatus } from './styles';
 
const Enquiries = ({
	isLoading,
	isEditEnabled,
	isSurfaceEnabled,
	enquiries,
	handleEnquiryCardClick,
	trackTooltipInfo,
	trackConfirmedFFInfo,
	handleCopy,
	handleEdit,
	handleArchive,
	handleUnarchive,
}) => {
	if (isLoading) return <EnquiriesLoader count={4} />;
 
	if ((enquiries || []).length === 0) {
		return <NoData>No data found!</NoData>;
	}
 
	return (enquiries || []).map(enquiry => {
		const {
			id,
			key,
			archived,
			created_by,
			freight_charges_needed,
			incoterm,
			mode,
			name,
			status,
			confirmed_vendors_preview,
			vendors_quoted_count,
			shipment_type,
			vendors,
			enquiry_label,
			client_group = {},
			next_approvers = [],
		} = enquiry || {};
 
		const isConfirmed = status === 'Confirmed';
 
		const isNewEnquiry = enquiry_label && enquiry_label !== 'Old';
 
		const handleCopyHelper = event => {
			const copyFn = () => handleCopy(event, id);
 
			/* is FTL, LTL and enabled to user => then only allow copy */
			if (isSurfaceEnquiry(enquiry)) {
				if (isSurfaceEnabled) {
					copyFn();
				} else {
					event.stopPropagation();
				}
			} else {
				copyFn();
			}
		};
 
		return (
			<EnquiryCard key={`${id}`} onClick={() => handleEnquiryCardClick(key)}>
				<Row className="row-1" type="flex">
					<CardColumnFlexStart xl={6} lg={6} md={6} xs={12}>
						<QuoteName>
							<Tooltip title={name || 'N/A'}>{name || 'N/A'}</Tooltip>
						</QuoteName>
						{isNewEnquiry && (
							<Tooltip
								placement="topLeft"
								title={(
									<CenteredText>
										{enquiry_label === 'New'
											? 'This tag highlights all enquiries which were created in the last 24 hours.'
											: 'This tag highlights all enquiries whose bid time was updated in the last 24 hours.'}
									</CenteredText>
								)}
							>
								<NewRFQStatus>
									<span>{enquiry_label === 'New' ? 'NEW' : 'UPDATED'}</span>
								</NewRFQStatus>
							</Tooltip>
						)}
					</CardColumnFlexStart>
 
					<CardColumn xl={3} lg={3} md={6} xs={12}>
						<CreatedBy>
							<Tooltip title={`${(created_by || {}).name} (${(client_group || {}).name})` || 'N/A'}>
								{(created_by || {}).name || 'N/A'}
							</Tooltip>
						</CreatedBy>
					</CardColumn>
 
					<StatusCol xl={7} lg={7} md={6} xs={12}>
						{(status === 'Open' || status === 'Closed') && next_approvers?.length > 0 ? (
							<Tooltip
								placement="topLeft"
								title={<EnquiryApproversToolTip approvers={next_approvers} />}
							>
								<StatusBar
									ghost
									value="Awaiting Approval"
									type="progress"
									className="status-other"
								/>
							</Tooltip>
						) : (
							status !== 'Open' && (
								<StatusBar
									ghost
									value={status}
									type={GET_STATUS_TYPE(status)}
									className={`status ${isConfirmed ? 'status-confirmed' : 'status-other'}`}
								/>
							)
						)}
						{/* showing confirmed FF in a tooltip */}
						{isConfirmed && (
							<TooltipConfirmedFF
								confirmedVendors={confirmed_vendors_preview}
								onInfoMouseEnter={trackConfirmedFFInfo}
							/>
						)}
					</StatusCol>
 
					{freight_charges_needed ? (
						<>
							<CardColumn xl={4} lg={4} md={3} xs={6} className="border">
								<VendorQuotes>
									<span>
										<i className="fas fa-user" />
									</span>
									<span className="no-of-vendors">{vendors || 0} Vendors</span>
								</VendorQuotes>
							</CardColumn>
 
							<CardColumn xl={4} lg={4} md={3} xs={6} className="border">
								<VendorQuotes>
									<span>
										<i className="fas fa-envelope" />
									</span>
									<span className="no-of-messages">{vendors_quoted_count || 0} Quotes</span>
								</VendorQuotes>
							</CardColumn>
						</>
					) : (
						<FreightNotRequired xl={8} lg={8} md={6} xs={12}>
							Freight Not Required
						</FreightNotRequired>
					)}
				</Row>
 
				<RowTwo type="flex" className="row-2">
					<CardColumn xl={6} lg={6} md={12} sm={12} xs={24} className="no-pad-r">
						<ShipmentInfo mode={mode}>
							<span className="icons shipment" />
							<div>
								<p>
									{shipment_type} {!isDomesticSurfaceEnquiry(enquiry) && <>({incoterm})</>} {mode}
									<Tooltip
										placement="right"
										title={<EnquiryTooltip enquiry={enquiry} userType="client" />}
									>
										<i className="fas fa-info-circle m-l-5" onMouseEnter={trackTooltipInfo} />
									</Tooltip>
								</p>
								<p>GID: {key}</p>
							</div>
						</ShipmentInfo>
					</CardColumn>
 
					<CardColumn xl={10} lg={10} md={12} sm={12} xs={24}>
						<LocationAddress details={enquiry} />
					</CardColumn>
 
					<CardColumn xl={4} lg={4} md={12} xs={12} className="no-pad-r">
						<BidCloseTime>{getListBiddingTime(enquiry)}</BidCloseTime>
					</CardColumn>
 
					{isEditEnabled && (
						<CardColumn xl={4} lg={4} md={12} xs={12}>
							<EditIcons>
								<Tooltip title="Copy">
									<span
										className={`icon-1 ${
											isEnquiryCopyEnabled(isSurfaceEnabled, enquiry) ? '' : 'disabled'
										}`}
										onKeyPress={() => {}}
										role="button"
										tabIndex={-1}
										onClick={handleCopyHelper}
									>
										<i className="far fa-copy" />
									</span>
								</Tooltip>
 
								<Tooltip title="Edit">
									<span
										className="icon-2"
										onKeyPress={() => {}}
										role="button"
										tabIndex={-1}
										onClick={event => handleEdit(event, id)}
									>
										<i className="fas fa-pencil-alt" />
									</span>
								</Tooltip>
 
								{!archived ? (
									<Tooltip title="Archive">
										<span
											role="button"
											tabIndex={0}
											onKeyDown={() => {}}
											className="icon-3"
											onClick={event => handleArchive(event, id)}
										>
											<Archive />
										</span>
									</Tooltip>
								) : (
									<>
										{isEnquiryCancelled(enquiry) ? (
											<Tooltip title="Unarchive">
												<span
													role="button"
													className="icon-3 disabled"
													tabIndex={0}
													onKeyDown={() => {}}
													onClick={event => event.stopPropagation()}
												>
													<Unarchive disabled />
												</span>
											</Tooltip>
										) : (
											<Tooltip title="Unarchive">
												<span
													role="button"
													tabIndex={0}
													onKeyDown={() => {}}
													className="icon-3"
													onClick={event => handleUnarchive(event, id)}
												>
													<Unarchive />
												</span>
											</Tooltip>
										)}
									</>
								)}
							</EditIcons>
						</CardColumn>
					)}
				</RowTwo>
			</EnquiryCard>
		);
	});
};
 
Enquiries.propTypes = {
	isLoading: PropTypes.boolean,
	isEditEnabled: PropTypes.bool,
	enquiryTypeValue: PropTypes.string,
	enquiries: PropTypes.shape([]),
	handleEnquiryCardClick: PropTypes.func,
	trackTooltipInfo: PropTypes.func,
	trackConfirmedFFInfo: PropTypes.func,
	handleCopy: PropTypes.func,
	handleEdit: PropTypes.func,
	handleArchive: PropTypes.func,
	handleUnarchive: PropTypes.func,
};
 
Enquiries.defaultProps = {
	isLoading: false,
	isEditEnabled: false,
	enquiryTypeValue: '',
	enquiries: [],
	handleEnquiryCardClick: () => {},
	trackTooltipInfo: () => {},
	trackConfirmedFFInfo: () => {},
	handleCopy: () => {},
	handleEdit: () => {},
	handleArchive: () => {},
	handleUnarchive: () => {},
};
 
export default Enquiries;
