import propTypes from 'prop-types';
import isArray from 'lodash/isArray';
import isObject from 'lodash/isObject';
import { isNil } from 'lodash';
import RoundNumber from 'common-util/round-number';
import { omitSingle } from 'common-util/reusable-functions';
import { keyToTitle } from '../../common';
 
/* List of keys whose values are needed to be rounded */
const roundedNumberKeys = [
	'amount_without_tax',
	'amount_with_tax',
	'charge_per_weight',
	'chargeable_weight',
	'volumetric_weight',
	'gross_weight',
];
 
const renderFromArray = data => (data || []).map(item => (Object.entries(item || {}) || []).map(([key, value]) => (
	<tr key={key}>
		<td>{keyToTitle(key)}</td>
		<td>
			{roundedNumberKeys.includes(key) && !isNil(value) ? (
				<RoundNumber isCurrencyVisible={false} amount={value} tooltipText={value} />
			) : (
				value
			)}
		</td>
	</tr>
)));
 
const renderFromObject = (key, object) => (
	<tr key={key}>
		<td>{keyToTitle(key)}</td>
		<td>{(Object.keys(object) || []).map(name => `${name}: ${object[name]}`)}</td>
	</tr>
);
 
const sortDetailsToPriority = (detail1, detail2) => {
	if (detail1.priority < detail2.priority) {
		return -1;
	}
	if (detail1.priority > detail2.priority) {
		return 1;
	}
	return 0;
};
 
/** This func do the following things
 * arranges the details order according to priority
 * removes the priority key to prevent hash order inconsistency
 */
const sortAndRemovePriority = details => {
	const sortedDetails = (details || []).sort(sortDetailsToPriority);
	return (sortedDetails || []).map(detail => omitSingle('priority', detail));
};
 
const Details = ({ invoice }) => {
	const sortedDetails = sortAndRemovePriority(invoice.details || []).sort(sortDetailsToPriority);
	return (sortedDetails || []).map(detailHash => {
		const detail = (Object.entries(detailHash) || [])[0];
		let key = detail[0];
		const value = detail[1];
 
		if (invoice.details.mode === 'LCL' && key === 'volumetric_weight') {
			key = 'volume';
		}
		if (isArray(value)) {
			return renderFromArray(value);
		}
		if (isObject(value)) {
			return renderFromObject(key, value);
		}
 
		return (
			<tr key={key}>
				<td>{keyToTitle(key)}</td>
				<td>
					{roundedNumberKeys.includes(key) && !isNil(value) ? (
						<RoundNumber isCurrencyVisible={false} amount={value} tooltipText={value} />
					) : (
						value
					)}
				</td>
			</tr>
		);
	});
};
 
Details.propTypes = {
	invoice: propTypes.shape({}),
	mode: propTypes.string,
};
 
Details.defaultProps = {
	invoice: {},
	mode: '',
};
 
export default Details;
