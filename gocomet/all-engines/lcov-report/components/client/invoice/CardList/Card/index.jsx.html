import PropTypes from 'prop-types';
import {
	Button, Icon, Row, Switch, Tooltip,
} from 'antd';
import { Location } from 'common-util/product/tracking';
import { getTotalDifference } from 'common-util/product/invoice';
import { InvoiceCard } from 'common-util/product/invoice/styles';
import InvoiceTooltip from 'common-util/product/invoice/components/InvoiceTooltip';
import {
	Ellipsis,
	RowTwo,
	CardColumn,
	ShipmentInfo,
	ShipmentLocation,
	EditIcons,
} from 'common-util/reusable-styles';
import { analyticsTrack } from 'util/mixpanel';
import { Archive, Unarchive } from 'common-util/icons/archive-unarchive';
import { StatusBar } from 'common-util/status-bar';
import React from 'react';
import { getStatusBar, InvoiceDocCount, getInvoiceStatus } from '../../common';
import {
	ErrorIcon,
	NextApproverContainer,
	ClickableDiv,
	FlexContainer,
	Dates,
	Date,
} from './styles';
import { NewStatusContainer } from '../styles';
 
const Card = ({
	invoiceData,
	viewInvoiceDetails,
	archiveInvoice,
	unarchiveInvoice,
	cancelInvoice,
	toggleAdditionalInvoice,
	handleShowAdditionalInvoiceModal,
	dispatchCard,
	vendor,
	additionInvoiceEnabled,
	permissions,
}) => {
	const {
		id,
		total_difference,
		client_verification_status,
		documents,
		archived,
		dispatch_data,
		parsed_data,
		error,
		form_requisites,
		auto_approved,
		auto_rejected,
		reuploaded_at,
		created_at,
		current_approvers,
		hide_comparison,
		newly_uploaded,
		can_upload_additional_invoice,
		is_additional_invoice,
	} = invoiceData || {};
 
	const { client_group: requisiteClientGroup } = form_requisites || {};
	const { client_configuration } = requisiteClientGroup || {};
	const { invoice } = client_configuration || {};
	const { details } = parsed_data || {};
	const { invoice_names } = details || {};
	const {
		enquiry_name = '',
		mode,
		vendor_group,
		vendor_types,
		key,
		pol,
		pod,
		currency,
		origin = '- -',
		destination = '- -',
	} = (dispatch_data || {}).details || {};
 
	const { company_name } = requisiteClientGroup || {};
	const { name: vendorName } = vendor_group || {};
	const getErrorValue = () => {
		const returnError = {};
		if ((error || {}).freight_charges_invoice) {
			returnError.freight_charges_invoice = (error || {}).freight_charges_invoice;
		}
		if ((error || {}).awb_charges_invoice) {
			returnError.awb_charges_invoice = (error || {}).awb_charges_invoice;
		}
		if ((error || {}).bl_charges_invoice) {
			returnError.bl_charges_invoice = (error || {}).bl_charges_invoice;
		}
 
		return returnError;
	};
 
	const errorToShow = Object.values(getErrorValue()).join(', ');
	const modeToShow = (mode || '').toUpperCase();
	const ellipsesStyle = { marginRight: '0.25rem', width: 'auto' };
	const canceled = client_verification_status === 'canceled';
	const approved = client_verification_status === 'approved';
	const handleViewInvoice = invoiceId => {
		analyticsTrack(`Invoice Card ${dispatchCard ? 'Dispatch List' : 'Invoice List'}`);
		viewInvoiceDetails(invoiceId);
	};
 
	const isInvoiceEditEnabled = permissions?.invoice?.invoice?.edit;
	const additionalInvoiceToggleVisible =		dispatchCard && additionInvoiceEnabled && approved && !is_additional_invoice;
	const columnWidth = additionalInvoiceToggleVisible ? 4 : null;
 
	// TODO: mixpanel event not working with id on div, need to fix this issue
 
	return (
		<ClickableDiv role="button" onClick={!canceled && (() => handleViewInvoice(id))}>
			<InvoiceCard className={canceled && 'disabled-card'}>
				<Row className="row-1">
					<CardColumn lg={columnWidth || 5} md={12} xs={24}>
						<Tooltip title={enquiry_name}>
							<Ellipsis style={ellipsesStyle} maxWidth={180}>
								{enquiry_name}
							</Ellipsis>
						</Tooltip>
						{newly_uploaded && (
							<NewStatusContainer>
								<span>New</span>
							</NewStatusContainer>
						)}
					</CardColumn>
 
					<CardColumn lg={columnWidth || 6} md={12} xs={24}>
						{!vendor && (
							<NextApproverContainer>
								Next approvers:&nbsp;
								<Tooltip title={`Next approvers: ${(current_approvers || ['']).join(', ')}`}>
									<Ellipsis display="inline" width={80}>
										{(current_approvers || []).join(', ')}
									</Ellipsis>
								</Tooltip>
							</NextApproverContainer>
						)}
					</CardColumn>
 
					<CardColumn lg={columnWidth || 5} md={12} xs={24}>
						<FlexContainer>
							{is_additional_invoice && <StatusBar value="Additional Invoice" type="primary" />}
							{getStatusBar(
								getInvoiceStatus(
									client_verification_status,
									auto_approved,
									reuploaded_at,
									auto_rejected,
								),
							)}
							{errorToShow && (
								<Tooltip title={`Error: ${errorToShow}`}>
									<ErrorIcon className="m-l-8 fas fa-info-circle error-text" />
								</Tooltip>
							)}
						</FlexContainer>
					</CardColumn>
 
					<CardColumn lg={columnWidth || 4} md={12} xs={12} className="border">
						{vendor ? (
							<Tooltip title={`Client: ${company_name}`}>
								<Ellipsis width={150}>Client: {company_name}</Ellipsis>
							</Tooltip>
						) : (
							<Tooltip title={`${vendor_types.join(', ')}: ${vendorName}`}>
								<Ellipsis width={150}>
									{vendor_types.join(', ')}: {vendorName}
								</Ellipsis>
							</Tooltip>
						)}
					</CardColumn>
 
					<CardColumn lg={columnWidth || 4} md={12} xs={12} className="border">
						<Tooltip title={`Invoice: ${(invoice_names || ['']).join(', ')}`}>
							<Ellipsis width={150}>Invoice: {(invoice_names || ['']).join(', ')}</Ellipsis>
						</Tooltip>
					</CardColumn>
					{additionalInvoiceToggleVisible && (
						<CardColumn lg={columnWidth || 4} md={12} xs={12}>
							<span>Additional Invoice</span>
						</CardColumn>
					)}
				</Row>
 
				<RowTwo type="flex" align="middle" className="row-2">
					<CardColumn xl={columnWidth || 5} md={12} sm={12} xs={24} className="no-pad-r">
						<ShipmentInfo mode={modeToShow}>
							<span className="icons shipment" />
							<div>
								<p>
									MODE: {modeToShow}
									<Tooltip placement="right" title={<InvoiceTooltip invoice={invoiceData} />}>
										<i className="fas fa-info-circle m-l-5" />
									</Tooltip>
								</p>
								<p>GID: {key}</p>
							</div>
						</ShipmentInfo>
					</CardColumn>
					<CardColumn lg={columnWidth || 6} md={12} xs={24}>
						{dispatchCard ? (
							<InvoiceDocCount
								documents={documents}
								mode={mode}
								awbBlRequired={(invoice || {}).awb_bl_document}
							/>
						) : (
							<ShipmentLocation>
								<div className="shipment-pol">
									{mode === 'ltl' || mode === 'ftl' ? (
										<p> {origin} </p>
									) : (
										<Location location={pol} />
									)}
								</div>
								<div className="shipment-pod">
									{mode === 'ltl' || mode === 'ftl' ? (
										<p> {destination} </p>
									) : (
										<Location location={pod} />
									)}
								</div>
							</ShipmentLocation>
						)}
					</CardColumn>
					<CardColumn lg={columnWidth || 5} md={12} xs={24} style={{ display: 'block' }}>
						<Dates>
							<Date>Uploaded At: {created_at}</Date>
							<Date style={reuploaded_at ? { visbility: 'visible' } : { visibility: 'hidden' }}>
								Reuploaded At: {reuploaded_at}
							</Date>
						</Dates>
					</CardColumn>
					<CardColumn lg={columnWidth || 4} md={12} xs={24}>
						{!hide_comparison && (
							<div style={{ display: 'grid' }}>
								<span>Difference</span>
								<>
									<Tooltip title={<>Difference {getTotalDifference(total_difference, currency)}</>}>
										<Ellipsis width={180}>
											{getTotalDifference(total_difference, currency)}
										</Ellipsis>
									</Tooltip>
								</>
							</div>
						)}
					</CardColumn>
					<CardColumn xl={4} lg={columnWidth || 4} md={12} xs={12}>
						{isInvoiceEditEnabled ? (
							<EditIcons className="editicons">
								{archived ? (
									<Tooltip title="UnArchive">
										<span
											role="button"
											tabIndex={0}
											onKeyDown={() => {}}
											id="invoice-unarchive-submit"
											data-testid="invoice-unarchive-submit"
											className="span-3 restore-btn"
											onClick={event => unarchiveInvoice(event, id)}
										>
											<Unarchive />
										</span>
									</Tooltip>
								) : (
									<Tooltip title="Archive">
										<span
											role="button"
											tabIndex={0}
											onKeyDown={() => {}}
											id="invoice-archive-submit"
											data-testid="invoice-archive-submit"
											className="span-3 archive-btn"
											onClick={event => archiveInvoice(event, id)}
										>
											<Archive />
										</span>
									</Tooltip>
								)}
								{!vendor && !canceled && (
									<Tooltip title="Cancel">
										<span
											role="presentation"
											className="icon-2"
											data-testid="invoice-cancel-submit"
											onClick={event => !approved && cancelInvoice(event, id)}
										>
											<i className={` far fa-trash-alt red ${approved && 'disabled'}`} />
										</span>
									</Tooltip>
								)}
							</EditIcons>
						) : null}
					</CardColumn>
					{additionalInvoiceToggleVisible && (
						<CardColumn xl={4} lg={columnWidth || 4} md={12} xs={12}>
							<EditIcons className="editicons">
								{!vendor && (
									<Switch
										checked={can_upload_additional_invoice}
										onChange={(checked, event) => toggleAdditionalInvoice(event, id)}
										className="m-r-8"
									/>
								)}
								{can_upload_additional_invoice && (
									<Tooltip title="Upload Additional invoice">
										<Button onClick={event => handleShowAdditionalInvoiceModal(event, id)}>
											<Icon type="upload" />
										</Button>
									</Tooltip>
								)}
							</EditIcons>
						</CardColumn>
					)}
				</RowTwo>
			</InvoiceCard>
		</ClickableDiv>
	);
};
 
Card.propTypes = {
	dispatchCard: PropTypes.bool,
	vendor: PropTypes.bool,
	additionInvoiceEnabled: PropTypes.bool,
	invoiceData: PropTypes.shape({}),
	permissions: PropTypes.shape({}),
	viewInvoiceDetails: PropTypes.func,
	archiveInvoice: PropTypes.func,
	unarchiveInvoice: PropTypes.func,
	toggleAdditionalInvoice: PropTypes.func,
	handleShowAdditionalInvoiceModal: PropTypes.func,
	cancelInvoice: PropTypes.func,
};
 
Card.defaultProps = {
	dispatchCard: false,
	vendor: false,
	additionInvoiceEnabled: false,
	invoiceData: {},
	permissions: {},
	viewInvoiceDetails: () => {},
	archiveInvoice: () => {},
	cancelInvoice: () => {},
	toggleAdditionalInvoice: () => {},
	handleShowAdditionalInvoiceModal: () => {},
	unarchiveInvoice: () => {},
};
 
export default Card;
