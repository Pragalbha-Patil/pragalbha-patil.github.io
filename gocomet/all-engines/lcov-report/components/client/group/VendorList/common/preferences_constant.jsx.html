export const getSelectedModes = modeArr => {
	const selectedModes = {
		fcl: false,
		lcl: false,
		isotank: false,
		flexitank: false,
		air: false,
		courier: false,
		bnbb: false,
		ftl: false,
		ltl: false,
		bcl: false,
	};
 
	(modeArr || []).forEach((modePref = '') => {
		if (modePref === 'fcl') selectedModes.fcl = true;
		if (modePref === 'lcl') selectedModes.lcl = true;
		if (modePref === 'isotank') selectedModes.isotank = true;
		if (modePref === 'flexitank') selectedModes.flexitank = true;
		if (modePref === 'air') selectedModes.air = true;
		if (modePref === 'bnbb') selectedModes.bnbb = true;
		if (modePref === 'ftl') selectedModes.ftl = true;
		if (modePref === 'ltl') selectedModes.ltl = true;
		if (modePref === 'bcl') selectedModes.bcl = true;
	});
 
	return selectedModes;
};

const getActiveLables = (options, obj) => (options || {})
	.filter(option => {
		const { value } = option || {};
		return (obj || {})[value] || false;
	})
	.map(({ label }) => label);
