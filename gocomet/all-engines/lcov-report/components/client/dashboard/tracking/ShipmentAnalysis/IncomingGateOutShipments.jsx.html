/* global am4charts am4core */
import React, { Component } from 'react';
import PropTypes from 'prop-types';
import { connect } from 'react-redux';
import isEqual from 'lodash/isEqual';
import { Button, Select, Tooltip } from 'antd';
import { triggerAnalyticsEvent } from 'util/mixpanel/mixpanel-actions';
import Loader from 'common-util/loader';
import { getHyphenWords } from '../../common';
import { checkGraph } from '../../common/graph-helper';
import { CardBody, CardHeader } from '../../common/styles';
import {
	DropdownH3,
	IncomingGateOutShipmentsContainer,
	IncomingGateOutShipmentsTable,
} from '../styles';
 
const { Option } = Select;
 
const shipmentsAtPortTableHeader = [
	{ title: 'Days at port', dataIndex: 'days' },
	{ title: 'Number of shipment', dataIndex: 'count' },
	{ title: 'No. of Containers', dataIndex: 'container_count' },
];
 
const incomingShipmentsTableHeader = [
	{ title: 'Weeks at port', dataIndex: 'week' },
	{ title: 'Arriving shipments', dataIndex: 'count' },
	{ title: 'No. of Containers', dataIndex: 'container_count' },
];
 
class IncomingGateOutShipments extends Component {
	state = {
		isTableViewVisible: false,
		isShipmentAtPort: false,
	};
 
	componentDidMount() {
		this.renderChart();
	}
 
	componentDidUpdate(prevProps, prevState) {
		const { isShipmentAtPort, isTableViewVisible } = this.state;
		const { masterFilter, incomingShipments, shipmentsAtPort } = this.props;
 
		if (
			(isTableViewVisible !== prevState.isTableViewVisible && !isTableViewVisible)
			|| isShipmentAtPort !== prevState.isShipmentAtPort
			|| !isEqual(prevProps.incomingShipments, incomingShipments)
			|| !isEqual(prevProps.shipmentsAtPort, shipmentsAtPort)
			|| !isEqual(prevProps.masterFilter, masterFilter)
		) {
			this.renderChart();
		}
	}
 
	onTableViewBtnClick = () => {
		this.setState(prevState => ({ isTableViewVisible: !prevState.isTableViewVisible }));
	};
 
	renderChart = () => {
		const { isShipmentAtPort } = this.state;
		const { incomingShipments, shipmentsAtPort } = this.props;
 
		if (checkGraph('incomingShipmentsDiv')) {
			return null;
		}
 
		// this ensures all charts are automatically disposed when orphaned.
		am4core.options.autoDispose = true;
		return am4core.createFromConfig(
			{
				data: isShipmentAtPort ? shipmentsAtPort : incomingShipments,
				xAxes: [
					{
						type: 'CategoryAxis',
						title: { text: isShipmentAtPort ? 'Days at Port' : '' },
						dataFields: { category: isShipmentAtPort ? 'days' : 'week' },
						renderer: { grid: { disabled: true } },
					},
				],
				yAxes: [
					{
						type: 'ValueAxis',
						title: { text: 'Number of Shipments' },
						cursorTooltipEnabled: false,
						min: 0,
					},
				],
				series: [
					{
						type: 'LineSeries',
						dataFields: {
							categoryX: isShipmentAtPort ? 'days' : 'week',
							valueY: 'count',
						},
						strokeWidth: 3,
						tooltip: {
							getFillFromObject: false,
							background: { fill: am4core.color('#67B7DC') },
							label: { fill: am4core.color('#000000') },
						},
						bullets: [
							{
								children: [
									{
										type: 'CircleBullet',
										strokeWidth: 8,
										stroke: am4core.color('#67B7DC70'),
										radius: 6,
										fill: am4core.color('#67B7DC'),
									},
								],
								tooltipText: 'Number of Shipments: {valueY}',
								states: { hover: { properties: { scale: 1.25 } } },
							},
						],
					},
				],
			},
			'incomingShipmentsDiv',
			am4charts.XYChart,
		);
	};
 
	handleChange = value => {
		this.setState({ isShipmentAtPort: value === 'shipments_at_port_key' });
	};
 
	trackTooltipInfo = () => {
		triggerAnalyticsEvent('tracking-dash-tooltip-incoming-shipments');
	};
 
	render() {
		const { isTableViewVisible, isShipmentAtPort } = this.state;
		const {
			isLoading, incomingShipments, shipmentsAtPort, gateOutEnabled,
		} = this.props;
 
		return (
			<IncomingGateOutShipmentsContainer>
				<CardHeader>
					<DropdownH3>
						<Select
							style={{ width: 220 }}
							value={isShipmentAtPort ? 'shipments_at_port_key' : 'incoming_shipments_key'}
							onChange={this.handleChange}
						>
							<Option value="incoming_shipments_key">Incoming shipments</Option>
							{gateOutEnabled && (
								<Option value="shipments_at_port_key">Shipments at port</Option>
							)}
						</Select>
 
						<Tooltip
							placement="right"
							title={
								isShipmentAtPort
									? 'Analyse the number of shipments waiting on Destination Port'
									: 'Analyse the number of shipments arriving at Destination Port within the set week'
							}
						>
							<i
								className="fas fa-info-circle"
								id="tracking-dash-tooltip-incoming-shipments"
								data-analytics-event-name="Info Tooltip"
								data-analytics-key-type="Shipments at port & Incoming Containers"
								onMouseEnter={this.trackTooltipInfo}
							/>
						</Tooltip>
					</DropdownH3>
 
					<div>
						<Button
							size="small"
							type="primary"
							id="tracking-dash-vt-incoming-shipments"
							data-analytics-event-name="View Toggle"
							data-analytics-key-type="Shipments at port & Incoming Containers"
							onClick={this.onTableViewBtnClick}
						>
							{isTableViewVisible ? 'Graph View' : 'Table View'}
						</Button>
					</div>
				</CardHeader>
 
				<CardBody className="card-body">
					{isLoading ? (
						<Loader />
					) : (
						<>
							{isTableViewVisible ? (
								<IncomingGateOutShipmentsTable
									columns={
										isShipmentAtPort ? shipmentsAtPortTableHeader : incomingShipmentsTableHeader
									}
									dataSource={isShipmentAtPort ? shipmentsAtPort : incomingShipments}
									bordered
									rowKey={
										isShipmentAtPort
											? ({ days }) => getHyphenWords('incoming-shipments-tr', days)
											: ({ week }) => getHyphenWords('incoming-shipments-tr', week)
									}
									pagination={false}
								/>
							) : (
								<div id="incomingShipmentsDiv" />
							)}
						</>
					)}
				</CardBody>
			</IncomingGateOutShipmentsContainer>
		);
	}
}
 
IncomingGateOutShipments.propTypes = {
	isLoading: PropTypes.bool,
	gateOutEnabled: PropTypes.bool,
	masterFilter: PropTypes.shape({}),
	shipmentsAtPort: PropTypes.objectOf(PropTypes.shape([])),
	incomingShipments: PropTypes.objectOf(PropTypes.shape([])),
};
 
IncomingGateOutShipments.defaultProps = {
	isLoading: true,
	gateOutEnabled: true,
	masterFilter: {},
	shipmentsAtPort: [],
	incomingShipments: [],
};
 
const mapStateToProps = state => {
	const { masterFilter, section_shipment_analysis } = state.dashboard.tracking || {};
	const { on_port_and_incoming_shipments } = section_shipment_analysis || {};
	const { shipments_on_port, incoming_shipments } = on_port_and_incoming_shipments || {};
 
	return {
		masterFilter,
		shipmentsAtPort: shipments_on_port,
		incomingShipments: incoming_shipments,
	};
};
 
export default connect(
	mapStateToProps,
	{},
)(IncomingGateOutShipments);
