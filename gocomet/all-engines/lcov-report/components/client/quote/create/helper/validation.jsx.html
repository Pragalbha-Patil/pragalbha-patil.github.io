import { upperCase, get, isEmpty } from 'lodash';
import {
	isModeTruck,
	isFCL,
	isLCL,
	isFTL,
	isLTL,
	isProductValid,
	SECTION_VALIDATION_NAMES,
	isRailFCL,
	isRailLCL,
} from './index';
 
export const checkCargoType = ({ hazardous, refrigerated, envirotainer }) => {
	if (hazardous && refrigerated) return 'Reefer Haz Cargo';
	if (hazardous && !refrigerated) return 'Haz Cargo';
	if (!hazardous && refrigerated) return 'Reefer Cargo';
	if (envirotainer) return 'Envirotainer Cargo';
	return 'General Cargo';
};
 
export const checkValidity = (type, enquiry, config) => {
	const {
		name,
		confirmed_quote_id,
		mode,
		incoterm,
		temperature,
		weight_type,
		bid_close_time,
		packages = [],
		containers = [],
		cargo_type = {},
		total_quantity = {},
		origin_charges = {},
		destination_charges = {},
		carrier = {},
		pol = {},
		pod = {},
		service = '',
		origin = '',
		origin_zip_code = '',
		destination = '',
		destination_zip_code = '',
		stuffing_location = '',
		destuffing_location = '',
		hbl = '',
		mbl = '',
		eta = '',
		etd = '',
		trucks = [],
		products,
		consignees,
	} = enquiry || {};
	const {
		quantity, net_weight, gross_weight, volumetric_weight,
	} = total_quantity || {};
	const { isProductRequiredForHaz, isDispatchFromEnquiry } = config;
	const mandatoryFields = get(config, 'mandatoryFields', {});
 
	const checkmode = () => {
		let isTemperatureValid = true;
 
		if (
			(checkCargoType(cargo_type) === 'Reefer Haz Cargo'
				|| checkCargoType(cargo_type) === 'Reefer Cargo'
				|| checkCargoType(cargo_type) === 'Envirotainer Cargo')
			&& !temperature
		) {
			isTemperatureValid = false;
		}
 
		if (isFCL(mode) || isRailFCL(mode)) {
			return mode && isTemperatureValid && containers.length !== 0 ? 'checked' : 'default-checked';
		}
 
		if (isFTL(mode)) {
			return mode && isTemperatureValid && trucks.length !== 0 ? 'checked' : 'default-checked';
		}
 
		return mode && isTemperatureValid ? 'checked' : 'default-checked';
	};
 
	const checkWeight = () => {
		if (isFCL(mode) || isRailFCL(mode)) {
			return containers.length > 0 ? 'checked' : 'default-checked';
		}
		if (isFTL(mode)) {
			const areTrucksValid = trucks.some(
				({ quantity: truckQuantity, truck_size }) => truckQuantity && truck_size,
			);
			return areTrucksValid ? 'checked' : 'default-checked';
		}
		if (isLTL(mode)) return 'checked';
		if (isLCL(mode) || isRailLCL(mode)) return 'checked';
		if (weight_type === 'total_shipment') {
			return quantity && gross_weight && volumetric_weight && net_weight ? 'checked' : 'empty';
		}
 
		const isValidPackage = packages.every(item => {
			const {
				count, length, width, height, weight,
			} = item;
			return count > 0 && length > 0 && width > 0 && height > 0 && weight > 0;
		});
 
		return packages.length > 0 && isValidPackage ? 'checked' : 'empty';
	};
 
	const checkService = () => {
		if (isModeTruck(mode)) {
			return origin && origin_zip_code && destination && destination_zip_code
				? 'checked'
				: 'default-checked';
		}
 
		let isOrginAddressValid = false;
		let isDestAddressValid = false;
		let isStuffingLocValid = false;
		let isDestuffingLocValid = false;
 
		if ((service === 'dtp' || service === 'dtd') && !origin) isOrginAddressValid = true;
		if ((service === 'ptd' || service === 'dtd') && !destination) isDestAddressValid = true;
		if (mode === 'fcl' && incoterm) {
			if ((origin_charges.cfs || origin_charges.port) && !stuffing_location) {
				isStuffingLocValid = true;
			}
			if ((destination_charges.cfs || destination_charges.port) && !destuffing_location) {
				isDestuffingLocValid = true;
			}
		}
 
		return incoterm
			&& (pol || {}).display_name
			&& (pod || {}).display_name
			&& !isOrginAddressValid
			&& !isDestAddressValid
			&& !isStuffingLocValid
			&& !isDestuffingLocValid
			? 'checked'
			: 'empty';
	};
 
	const checkConsignmentDetails = () => {
		if (
			(mandatoryFields.shipper && isEmpty(get(consignees, 'shipper_group.id', '')))
			|| (mandatoryFields.consignee && isEmpty(get(consignees, 'consignee_group.id', '')))
			|| (mandatoryFields.customer && isEmpty(get(consignees, 'customer_group.id', '')))
		) {
			return 'default-checked';
		}
		return 'checked';
	};
 
	switch (type) {
		case SECTION_VALIDATION_NAMES.reference.value:
			if (isDispatchFromEnquiry) {
				return name && confirmed_quote_id ? 'checked' : 'default-checked';
			}
			return name ? 'checked' : 'default-checked';
		case SECTION_VALIDATION_NAMES.consignment.value:
			return checkConsignmentDetails();
		case SECTION_VALIDATION_NAMES.mode.value:
			return checkmode();
		case SECTION_VALIDATION_NAMES.weightvolume.value:
			return checkWeight();
		case SECTION_VALIDATION_NAMES.addresservice.value:
			return checkService();
		case SECTION_VALIDATION_NAMES.product.value: {
			if (isProductValid(mandatoryFields.product, isProductRequiredForHaz, cargo_type, products)) {
				return 'checked';
			}
			return 'default-checked';
		}
		case SECTION_VALIDATION_NAMES.otherdetails.value:
			return name && checkmode() === 'checked' && checkWeight() === 'checked'
				? 'checked'
				: 'default-checked';
		case SECTION_VALIDATION_NAMES.enquiryschedule.value:
			return bid_close_time ? 'checked' : 'empty';
 
		/* below are custom fields, hence already checked */
		case SECTION_VALIDATION_NAMES.dispatchcustomfields.value:
			return hbl || mbl || eta || etd || !isEmpty(carrier) ? 'checked' : 'default-checked';
		case SECTION_VALIDATION_NAMES.freighceilingrate.value:
			return 'checked';
		case SECTION_VALIDATION_NAMES.customcarrier.value:
			return 'checked';
		case SECTION_VALIDATION_NAMES.slabrate.value:
			return name && checkmode() === 'checked' && checkWeight() === 'checked'
				? 'checked'
				: 'default-checked';
		default:
			return 'empty';
	}
};
 
export const checkIncotermValidity = (
	incoterm = '',
	shipment_type = '',
	pickDrop = '',
	service_name,
) => {
	if (shipment_type === 'export') {
		if (
			['CIF', 'CNF', 'CPT', 'CIP', 'CFR'].includes(upperCase(incoterm))
			&& ['DTD', 'PTD'].includes(upperCase(pickDrop))
		) {
			return `You have selected: ${shipment_type} ${upperCase(
				incoterm,
			)} with ${service_name} service`;
		}
		if (
			['EXW', 'FOB', 'FCA', 'FAS'].includes(upperCase(incoterm))
			&& ['PTP', 'PTD', 'DTP', 'DTD'].includes(upperCase(pickDrop))
		) {
			return `You have selected: ${shipment_type} ${upperCase(
				incoterm,
			)} with ${service_name} service`;
		}
		if (['DDU', 'DDP'].includes(upperCase(incoterm)) && ['DTP', 'PTP'].includes(pickDrop)) {
			return `You have selected: ${shipment_type} ${incoterm} with ${service_name} service`;
		}
		return null;
	}
 
	/* Import */
	if (
		['CIF', 'CNF', 'CPT', 'CIP', 'CFR'].includes(upperCase(incoterm))
		&& ['DTP', 'DTD'].includes(upperCase(pickDrop))
	) {
		return `You have selected: ${shipment_type} ${upperCase(
			incoterm,
		)} with ${service_name} service`;
	}
 
	if (
		['FOB', 'FCA', 'FAS'].includes(upperCase(incoterm))
		&& ['DTP', 'DTD'].includes(upperCase(pickDrop))
	) {
		return `You have selected: ${shipment_type} ${upperCase(
			incoterm,
		)} with ${service_name} service`;
	}
 
	return null;
};
