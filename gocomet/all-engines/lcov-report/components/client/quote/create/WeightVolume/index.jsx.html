import React, { Component } from 'react';
import PropTypes from 'prop-types';
import {
	Tabs, Checkbox, Row, Col,
} from 'antd';
import { get, isNumber } from 'lodash';
import { Section, Title } from 'common-util/product/forms/styles';
import {
	isFCL,
	isLCL,
	isBCL,
	isFTL,
	isLTL,
	isAir,
	isRailFCL,
	isRailLCL,
	isWarehouse,
	getUpdatedTrucksAndQuantity,
	getUpdatedTotalQuantity,
} from '../helper';
import ShipmentFCL from './ByShipmentFCL';
import ShipmentLclOrAir from './ByShipmentLclAir';
import ByShipmentFTL from './ByShipmentFTL';
import Packaging from './ByPackaging';
import {
	GET_DEFAULT_PACKAGE,
	getWeights,
	getPackages,
	getLengthUnits,
	NoModeError,
	weightVolumeError,
	getStackType,
} from './helper';
import { Box, InputNumeric } from '../styles';
import { WeightVolumeTabs, Palletization } from './styles';
 
const { TabPane } = Tabs;
 
class WeightVolume extends Component {
	onWeightVolumeTabChange = key => {
		const { handler } = this.props;
		handler('weight_type', key);
	};
 
	/* ---------------- common helpers ---------------- */
	onSelect = (name, value) => {
		const { handler, data } = this.props;
		const { total_quantity = {} } = data || {};
		total_quantity[name] = value;
		handler('total_quantity', total_quantity);
	};
 
	/* ---------------- FCL helpers ---------------- */
	handleFCLContainers = (index, name, value) => {
		const { handler, data } = this.props;
		const { containers, total_quantity } = data || {};
 
		const containersCopy = [...(containers || [])];
		(containersCopy || [])[index][name] = value;
		const totalQuantityCopy = getUpdatedTotalQuantity(containersCopy, total_quantity);
 
		handler('containers', containersCopy);
		handler('total_quantity', totalQuantityCopy);
	};
 
	onInputChange = (value, name) => {
		const { handler, data } = this.props;
		const { total_quantity } = data || {};
		const totalQuantityCopy = {
			...(total_quantity || {}),
			[name]: value,
		};
		handler('total_quantity', totalQuantityCopy);
	};
 
	/* ---------------- FTL helpers ---------------- */
	handleTruckValueChange = (index, name, value) => {
		const { handler, data } = this.props;
		const { trucks, total_quantity } = data || {};
		const { updatedTrucks, updatedTotalQuantity } = getUpdatedTrucksAndQuantity(
			index,
			name,
			value,
			trucks,
			total_quantity,
		);
		handler('trucks', updatedTrucks);
		handler('total_quantity', updatedTotalQuantity);
	};
 
	/* ---------------- LCL helpers ---------------- */
	onShipmentLclAirSelect = (name, value) => {
		const { handler, data } = this.props;
		const { total_quantity, mode } = data || {};
 
		if (mode === 'air' && name !== 'package_type') {
			total_quantity.weight_unit = value;
			total_quantity.volume_unit = value;
		} else {
			total_quantity[name] = value;
		}
		handler('total_quantity', total_quantity);
	};
 
	/* ---------------- packaging tab ---------------- */
	handlePackage = (index, name, value) => {
		const { handler, data } = this.props;
		const { packages = [] } = data || {};
		const packagesCopy = [...packages];
		(packagesCopy || [])[index][name] = value;
		handler('packages', packagesCopy);
	};
 
	handleAddMorePackaging = () => {
		const { handler, data, enquiryConfigurations } = this.props;
		const { packages } = data || {};
		const packagesCopy = [...(packages || [])];
		const configurations = {
			default_weight_unit: get(
				enquiryConfigurations,
				'account_preferences.default_units.weight_unit',
			),
			default_length_unit: get(
				enquiryConfigurations,
				'account_preferences.default_units.length_unit',
			),
		};
		packagesCopy.push({ ...GET_DEFAULT_PACKAGE(configurations || {}) });
		handler('packages', packagesCopy);
	};
 
	handlePackagingDelete = index => {
		const { handler, data = {} } = this.props;
		const { packages = [] } = data || {};
		const packagesCopy = [...packages];
		packagesCopy.splice(index, 1);
		handler('packages', packagesCopy);
	};
 
	/* ---------------- palletization ---------------- */
	handlePalletization = (name, value) => {
		const { handler, data } = this.props;
		const palletizationCopy = { ...((data || {}).palletization_charges || {}) };
		if (name === 'pallet_count') {
			palletizationCopy.pallet_count = value || null;
		}
		if (name === 'required') {
			palletizationCopy.required = value;
		}
		handler('palletization_charges', palletizationCopy);
	};
 
	/* ---------------- RENDER ---------------- */
	render() {
		const {
			data,
			isPalletizationReq,
			isSectionDisabled,
			activeMenu,
			formRequisites,
			handleUploadDocs,
			handleRemoveDocs,
		} = this.props;
		const {
			weight_type = 'by_package',
			total_quantity = {},
			packages = [],
			containers = [],
			mode,
			trucks,
			palletization_charges,
			documents,
		} = data || {};
		const { weight_unit } = total_quantity || {};
 
		const weightUnits = getWeights(formRequisites);
		const packageUnits = getPackages(formRequisites);
		const stackTypes = getStackType(formRequisites);
		const lengthUnits = getLengthUnits(formRequisites);
 
		/* palletization */
		const { required, pallet_count } = palletization_charges || {};
 
		return (
			<Section id="weight" className={activeMenu === 'weight' ? 'active' : null}>
				<Title>Weight & Volume</Title>
				<Box className={weightVolumeError(data) ? 'error-border' : null}>
					<WeightVolumeTabs defaultActiveKey={weight_type} onChange={this.onWeightVolumeTabChange}>
						<TabPane tab="By Total Shipment" key="total_shipment">
							{!mode && <NoModeError />}
 
							{(isFCL(mode) || isRailFCL(mode)) && (
								<ShipmentFCL
									isSectionDisabled={isSectionDisabled}
									containers={containers}
									weightUnit={weight_unit}
									weightUnits={weightUnits}
									handleFCLContainers={this.handleFCLContainers}
									onSelect={this.onSelect}
								/>
							)}
 
							{isFTL(mode) && (
								<ByShipmentFTL
									isSectionDisabled={isSectionDisabled}
									trucks={trucks}
									weightUnit={weight_unit}
									weightUnits={weightUnits}
									handleTruckValueChange={this.handleTruckValueChange}
									onTruckTypeSelect={this.onSelect}
								/>
							)}
 
							{(isLCL(mode)
								|| isLTL(mode)
								|| isRailLCL(mode)
								|| isAir(mode)
								|| isWarehouse(mode)
								|| isBCL(mode)) && (
								<ShipmentLclOrAir
									isSectionDisabled={isSectionDisabled}
									mode={mode}
									weightUnits={weightUnits}
									packageTypes={packageUnits}
									stackTypes={stackTypes}
									totalQuantity={total_quantity}
									onInputChange={this.onInputChange}
									onSelect={this.onShipmentLclAirSelect}
								/>
							)}
						</TabPane>
 
						<TabPane tab="By Packaging" key="by_package">
							<Packaging
								isSectionDisabled={isSectionDisabled}
								mode={mode}
								packages={packages}
								packageUnits={packageUnits}
								weightUnits={weightUnits}
								lengthUnits={lengthUnits}
								stackTypes={stackTypes}
								documents={documents}
								handleDelete={this.handlePackagingDelete}
								handleAddMore={this.handleAddMorePackaging}
								handlePackage={this.handlePackage}
								onUpload={handleUploadDocs}
								handleRemoveDocs={handleRemoveDocs}
							/>
						</TabPane>
					</WeightVolumeTabs>
				</Box>
				{weightVolumeError(data) && (
					<div className="ant-form-item-control has-error">
						<div className="ant-form-explain">Please fill Weight & Volume</div>
					</div>
				)}
 
				{isPalletizationReq && isAir(mode) && (
					<Row>
						<Palletization>
							<Col xl={6} lg={8} md={10} sm={24} xs={24}>
								<Checkbox
									disabled={isSectionDisabled}
									checked={required}
									onChange={({ target }) => this.handlePalletization('required', target.checked)}
								>
									Palletization Charge Required
								</Checkbox>
							</Col>
 
							{required && (
								<Col xl={6} lg={10} md={10} sm={24} xs={24}>
									<span className="ant-form-item-label">No. of Pallets</span>
									<InputNumeric
										type="number"
										disabled={isSectionDisabled}
										value={isNumber(pallet_count) ? pallet_count : null}
										min={1}
										onChange={value => this.handlePalletization('pallet_count', value)}
									/>
								</Col>
							)}
						</Palletization>
					</Row>
				)}
			</Section>
		);
	}
}
 
WeightVolume.propTypes = {
	isPalletizationReq: PropTypes.bool,
	isSectionDisabled: PropTypes.bool,
	activeMenu: PropTypes.string,
	data: PropTypes.shape({}),
	enquiryConfigurations: PropTypes.shape({}),
	formRequisites: PropTypes.shape({}),
	handleUploadDocs: PropTypes.func,
	handleRemoveDocs: PropTypes.func,
	handler: PropTypes.func,
};
 
WeightVolume.defaultProps = {
	isPalletizationReq: false,
	isSectionDisabled: false,
	activeMenu: '',
	data: {} /* enquiry object */,
	enquiryConfigurations: {},
	formRequisites: {},
	handleUploadDocs: () => {},
	handleRemoveDocs: () => {},
	handler: () => {},
};
 
export default WeightVolume;
