import React from 'react';
import { connect } from 'react-redux';
import PropTypes from 'prop-types';
import {
	Timeline, Empty, Card, Tooltip,
} from 'antd';
import { ShadedButton } from 'common-util/buttons';
import { COLOR } from 'util/theme';
import { popFromStage, updateCurrentIndexes } from 'store/admin/reports-dashboard/actions';
import { HistoryHeader, Title, HistoryBody } from '../common/styles';
import { getOperationType, getOperationIcon } from './helpers';
import {
	Container, SmallerText, TimelineParent, FilterIcon, StageHeader,
} from './styles';
 
const SaveStack = ({
	stages, currentIndexes, popFromStageFn, updateCurrentIndexesFn,
}) => {
	const onFilterIconClick = stageIndex => {
		updateCurrentIndexesFn({ stageIndex });
	};
 
	const renderTimeline = operations => (operations || []).map((operation, index) => {
		const operationType = getOperationType(operation);
		const { title, displayName } = operationType || {};
 
		return (
			<Timeline.Item dot={getOperationIcon(operation)} key={index}>
				<TimelineParent>
					<div>
						<p>{title}</p>
						<SmallerText>{displayName || '--'}</SmallerText>
					</div>
				</TimelineParent>
			</Timeline.Item>
		);
	});
 
	const getStages = () => {
		const { stageIndex } = currentIndexes || {};
 
		const panelArr = [];
		(stages || []).forEach((stage, index) => {
			const { operations, filters } = stage;
			panelArr.push(
				<Card
					bordered={false}
					title={(
						<StageHeader>
							<div>{`Stage ${index + 1}`}</div>
							<div>
								<FilterIcon
									type="filter"
									theme={(filters || []).length === 0 ? 'outlined' : 'twoTone'}
									twoToneColor={COLOR.RED}
									className={index === stageIndex ? 'current-filter' : ''}
									onClick={() => {
										onFilterIconClick(index);
									}}
								/>
							</div>
						</StageHeader>
					)}
					key={`stage-${index}`}
				>
					{renderTimeline(operations)}
				</Card>,
			);
		});
 
		return <div>{panelArr}</div>;
	};
 
	return (
		<Container>
			<HistoryHeader>
				<Title>History</Title>
				<Tooltip placement="left" title="Remove last operation in stage">
					<ShadedButton
						size="small"
						disabled={(stages || []).length === 0}
						onClick={popFromStageFn}
					>
						Remove
					</ShadedButton>
				</Tooltip>
			</HistoryHeader>
 
			<HistoryBody>
				{(stages || []).length === 0 ? (
					<Empty
						image={Empty.PRESENTED_IMAGE_SIMPLE}
						description="Please use Collection, Group-By or Unwind to see stages"
						style={{ marginTop: '8px' }}
					/>
				) : (
					<>{getStages()}</>
				)}
			</HistoryBody>
		</Container>
	);
};
 
SaveStack.defaultProps = {
	// requestStatus: {},
	// messages: {},
	stages: [],
	currentIndexes: {},
	popFromStageFn: () => {},
	updateCurrentIndexesFn: () => {},
};
 
SaveStack.propTypes = {
	// requestStatus: PropTypes.objectOf(PropTypes.shape({})),
	// messages: PropTypes.objectOf(PropTypes.shape({})),
	stages: PropTypes.shape([]),
	currentIndexes: PropTypes.shape({}),
	popFromStageFn: PropTypes.func,
	updateCurrentIndexesFn: PropTypes.func,
};
 
const mapStateToProps = state => {
	const { currentSheet, currentIndexes } = (state.admin || {}).reportsDashboard || {};
	const { stages } = currentSheet || {};
	return {
		stages,
		currentIndexes,
	};
};
 
export default connect(
	mapStateToProps,
	{ popFromStageFn: popFromStage, updateCurrentIndexesFn: updateCurrentIndexes },
)(SaveStack);
