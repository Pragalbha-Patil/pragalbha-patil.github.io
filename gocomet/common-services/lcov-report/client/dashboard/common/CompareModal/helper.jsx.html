	const getContent = () => {
		if (productType === 'enquiry') {
			return (data || []).map(({ vendor }, index) => (
				<tr key={`${vendor}-${index + 1}`}>
					<td>{index + 1}</td>
					<td>{vendor}</td>
				</tr>
			));
		}
 
		if (productType === 'tracking') {
			return data.map(({ carrier_display_name, carrier_code }, index) => (
				<tr key={`${carrier_code}-${index + 1}`}>
					<td>{index + 1}.</td>
					<td>
						{carrier_display_name || ''} ({carrier_code || ''})
					</td>
				</tr>
			));
		}
 
		return null;
	};
 
	return (
		<TopFive>
			<div className="title">Top 5 {getName()}</div>
			{(data || []).length === 0 ? (
				<EmptyTopFive>No data</EmptyTopFive>
			) : (
				<div className="top-five-table">
					<table>
						<tbody>{getContent()}</tbody>
					</table>
				</div>
			)}
		</TopFive>
	);
};
