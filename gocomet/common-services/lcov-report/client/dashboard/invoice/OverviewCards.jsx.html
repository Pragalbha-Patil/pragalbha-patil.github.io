import React, { useState } from 'react';
import PropTypes from 'prop-types';
import { connect } from 'react-redux';
import { getInvoiceOverviewDetails as getInvoiceOverviewDetailsFn } from 'store/dashboard/invoice/actions';
import { OverviewInfo } from '../common';
import ModalLineGraph from '../common/ModalLineGraph';
import { InfoCards, SectionHeader } from '../common/styles';
 
const OverviewCards = ({
	isLoading,
	isPDFDownloading,
	section_overview,
	section_overview_details,
	client_currency,
	getInvoiceOverviewDetails,
}) => {
	const {
		no_of_invoices_uploaded = 'NA',
		no_of_pending_approval = 'NA',
		no_of_invoices_approved = 'NA',
		no_of_invoices_rejected = 'NA',
		total_savings_value = 'NA',
		total_value_processed_amount = 'NA',
	} = section_overview || {};
 
	const [isInvoiceModalVisible, toggleInvoiceModal] = useState(false);
	const [overviewType, updateOverviewType] = useState(null);
 
	const handleViewDetails = type => {
		toggleInvoiceModal(true);
		updateOverviewType(type);
		getInvoiceOverviewDetails(type);
	};
 
	const handleCancel = () => {
		toggleInvoiceModal(false);
	};
 
	const handleType = type => (type === 'total_savings' || type === 'total_value_processed'
		? 'total_savings_realization'
		: 'Invoice');
 
	return (
		<>
			<div className="dom-to-pdf">
				<SectionHeader id="invoice-overview">Overview</SectionHeader>
 
				<InfoCards isPDFDownloading={isPDFDownloading} cardsLength={6}>
					<OverviewInfo
						isLoading={isLoading}
						index={1}
						type="primary"
						value={no_of_invoices_uploaded}
						title="Invoices Uploaded"
						viewDetailsId="invoice-dashboard-overview-no-of-invoices-uploaded"
						viewDetailsHandler={() => handleViewDetails('invoices_uploaded')}
					/>
 
					<OverviewInfo
						isLoading={isLoading}
						index={2}
						type="warning"
						value={no_of_pending_approval}
						title="Pending for Approval"
						viewDetailsId="invoice-dashboard-overview-no-of-pending-approval"
						viewDetailsHandler={() => handleViewDetails('pending_approval')}
					/>
 
					<OverviewInfo
						isLoading={isLoading}
						index={3}
						type="success"
						value={no_of_invoices_approved}
						title="Approved Invoices"
						viewDetailsId="invoice-dashboard-overview-no-of-approved-invoices"
						viewDetailsHandler={() => handleViewDetails('invoices_approved')}
					/>
 
					<OverviewInfo
						isLoading={isLoading}
						index={4}
						type="danger"
						value={no_of_invoices_rejected}
						title="Rejected Invoices"
						viewDetailsId="invoice-dashboard-overview-no-of-rejected-invoices"
						viewDetailsHandler={() => handleViewDetails('invoices_rejected')}
					/>
 
					<OverviewInfo
						isLoading={isLoading}
						index={5}
						type="primary"
						value={total_value_processed_amount}
						title={`Total value processed (${client_currency || '--'})`}
						viewDetailsId="invoice-dashboard-overview-total-value-processed"
						viewDetailsHandler={() => handleViewDetails('total_value_processed')}
					/>
 
					<OverviewInfo
						isLoading={isLoading}
						index={6}
						type="warning"
						value={total_savings_value}
						title={`Total savings (${client_currency || '--'})`}
						viewDetailsId="invoice-dashboard-overview-total-savings"
						viewDetailsHandler={() => handleViewDetails('total_savings')}
					/>
				</InfoCards>
			</div>
 
			{isInvoiceModalVisible && (
				<ModalLineGraph
					isVisible={isInvoiceModalVisible}
					dashboardType={handleType(overviewType)}
					overviewType={overviewType}
					shipmentsData={section_overview_details}
					handleCancel={handleCancel}
				/>
			)}
		</>
	);
};
 
OverviewCards.propTypes = {
	isLoading: PropTypes.bool,
	section_overview: PropTypes.shape({}),
	section_overview_details: PropTypes.shape([]),
	client_currency: PropTypes.string,
};
 
OverviewCards.defaultProps = {
	isLoading: true,
	section_overview: {},
	section_overview_details: [],
	client_currency: '',
};
 
const mapStateToProps = state => {
	const { invoice } = state.dashboard || {};
	const { section_overview, section_overview_details, client_currency } = invoice || {};
 
	return { section_overview, section_overview_details, client_currency };
};
 
export default connect(
	mapStateToProps,
	{
		getInvoiceOverviewDetails: getInvoiceOverviewDetailsFn,
	},
)(OverviewCards);
