import PropTypes from 'prop-types';
import { Component } from 'react';
import { connect } from 'react-redux';
import get from 'lodash/get';
import { getMeta } from 'util/redux-utils';
import { getInvoiceExtraChargeHead } from 'store/dashboard/invoice/actions';
import Graph from '../../common/Graph';
 
const COLUMNS_HEADER = currency => [
	{
		title: 'Extra Charge Head',
		dataIndex: 'charge_head',
	},
	{
		title: `Total Expenditure (${currency || '--'})`,
		dataIndex: 'total_expenditure',
	},
];
 
class ExtraChargeHead extends Component {
	constructor(props) {
		super(props);
		const { masterFilter: { mode, extra_charge_head_mode } } = this.props;
		this.state = {
			subFilterModeType: mode || extra_charge_head_mode,
		};
	}
 
	static getDerivedStateFromProps(nextProps, prevState) {
		const { masterFilter: { mode, extra_charge_head_mode } } = nextProps;
		return {
			subFilterModeType: mode || extra_charge_head_mode,
		};
	}
 
	getAppliedFilters = () => {
		const { masterFilter } = this.props;
		const {
			created_at, cost_type, shipment_type, vendor_company_id, client_group_id,
		} =	masterFilter || {};
 
		const apiFilters = {
			mode: null,
			cost_type: cost_type || null,
			vendor_company_id: (vendor_company_id || []).length > 0 ? vendor_company_id : [''],
			client_group_id: (client_group_id || []).length > 0 ? client_group_id : [''],
			shipment_type: shipment_type || null,
			created_at: created_at || null,
		};
 
		return apiFilters;
	}
 
	handleSubFilterChange = value => {
		const { getInvoiceExtraChargeHeadFn } = this.props;
		const appliedFilters = this.getAppliedFilters();
		appliedFilters.extra_charge_head_mode = value || '';
		getInvoiceExtraChargeHeadFn(appliedFilters);
		this.setState({ subFilterModeType: value || '' });
	}
 
	render() {
		const {
			isLoading,
			isExtraChargeLoading,
			isSurfaceEnabled,
			masterFilter,
			tableData,
			graphData,
			currency,
		} = this.props;
 
		const { mode } = masterFilter || {};
 
		const { subFilterModeType } = this.state;
 
		const graphInformation = {
			type: 'bar',
			chartId: 'extraChargeHeadChart',
			graphData: graphData || [],
			xField: 'charge_head',
			xText: 'Charge Head',
			yField: 'instances',
			yText: 'No. of Instances',
			color: 'purple',
		};
 
		return (
			<Graph
				isLoading={isLoading || isExtraChargeLoading}
				location="right"
				id="extra-charge-head"
				title="Extra Charge Head"
				graphInformation={graphInformation}
				graphData={graphData}
				tableHeader={COLUMNS_HEADER(currency)}
				tableData={tableData}
				masterFilterModeType={mode}
				subFilterModeType={subFilterModeType}
				onSubFilterChange={this.handleSubFilterChange}
				isSurfaceEnabled={isSurfaceEnabled}
			/>
		);
	}
}
 
ExtraChargeHead.propTypes = {
	isLoading: PropTypes.bool,
	isExtraChargeLoading: PropTypes.bool,
	isSurfaceEnabled: PropTypes.bool,
	currency: PropTypes.string,
	graphData: PropTypes.shape([]),
	tableData: PropTypes.shape([]),
	masterFilter: PropTypes.objectOf(Object),
	getInvoiceExtraChargeHeadFn: PropTypes.func,
};
 
ExtraChargeHead.defaultProps = {
	isLoading: true,
	isExtraChargeLoading: false,
	isSurfaceEnabled: false,
	currency: '',
	graphData: [],
	tableData: [],
	masterFilter: {},
	getInvoiceExtraChargeHeadFn: () => {},
};
 
const mapStateToProps = state => {
	const {
		section_charge_head_analysis,
		client_currency,
		masterFilter,
	} = (state.dashboard || {}).invoice || {};
	const { extra_charge_head } = section_charge_head_analysis || {};
	const { graph, table } = extra_charge_head || {};
 
	const { isLoading: isExtraChargeLoading } = getMeta((state.dashboard || {}).invoice, 'getInvoiceExtraChargeHead') || {};
 
	const { configurations } = (state.users || {}).profileDetails || {};
	const { surface_transportation_module_enabled } = get(configurations, 'enquiry.auction', false);
 
 
	return {
		masterFilter,
		isExtraChargeLoading,
		currency: client_currency,
		graphData: graph || [],
		tableData: table || [],
		isSurfaceEnabled: surface_transportation_module_enabled,
	};
};
 
export default connect(
	mapStateToProps,
	{ getInvoiceExtraChargeHeadFn: getInvoiceExtraChargeHead },
)(ExtraChargeHead);
