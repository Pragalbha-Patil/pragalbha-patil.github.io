import { Component } from 'react';
import PropTypes from 'prop-types';
import { connect } from 'react-redux';
import {
	Table, Row, Col, Button, Menu, Dropdown, Icon,
} from 'antd';
import { analyticsTrack } from 'util/mixpanel';
import UploadExcel from 'common-util/upload-excel';
import Search from 'common-util/search';
import {
	GET_SELECTED_SERVICE_VALUES,
	GET_SELECTED_MODE_VALUES,
	GET_SELECTED_ROUTE_VALUES,
	GET_SELECTED_TRANSPORT_MODE_VALUES,
	GET_SELECTED_TRANSPORT_SERVICE_VALUES,
} from './common/preferences_constant';
import { CustomButtonGroup } from '../styles';
 
class ListTable extends Component {
	constructor(props) {
		super(props);
		const { vendorData } = props; /* adding vendorData to state */
		this.state = {
			searchValue: '',
			filteredTableData: vendorData,
			isUploadModalVisible: false,
		};
	}
 
	componentDidUpdate(prevProps) {
		const { vendorData } = prevProps;
		const { vendorData: propsClient } = this.props;
		if (vendorData !== propsClient) {
			// eslint-disable-next-line react/no-did-update-set-state
			this.setState({
				filteredTableData: propsClient,
			});
		}
	}
 
	showVendorUploadModal = () => {
		this.setState({ isUploadModalVisible: true });
	};
 
	hideUploadRateModal = eventId => {
		analyticsTrack('Vendor Bulk Upload Modal Close', { 'Step Closed On': eventId });
		this.setState({ isUploadModalVisible: false });
	};
 
	/* search by service (import, export, transport) Customs, Freight */
	isServicePresent = (isPresent, preferences) => {
		const { import: importDetails, export: exportDetails, domestic } = preferences || {};
		const isImpServicePresent = GET_SELECTED_SERVICE_VALUES((importDetails || {}).service).some(
			eachModePresent => isPresent(eachModePresent),
		);
		const isExpServicePresent = GET_SELECTED_SERVICE_VALUES((exportDetails || {}).service).some(
			eachModePresent => isPresent(eachModePresent),
		);
		const isTransServicePresent = GET_SELECTED_TRANSPORT_SERVICE_VALUES(
			(domestic || {}).service,
		).some(eachModePresent => isPresent(eachModePresent));
 
		return isImpServicePresent || isExpServicePresent || isTransServicePresent;
	};
 
	/* search by mode (import, export, transport) such as FTL, FCL, ISO tank, etc */
	isModePresent = (isPresent, preferences) => {
		const { import: importDetails, export: exportDetails, domestic } = preferences || {};
		const isImportModePresent = GET_SELECTED_MODE_VALUES((importDetails || {}).mode).some(
			eachModePresent => isPresent(eachModePresent),
		);
		const isExportModePresent = GET_SELECTED_MODE_VALUES((exportDetails || {}).mode).some(
			eachModePresent => isPresent(eachModePresent),
		);
		const isTransportModePresent = GET_SELECTED_TRANSPORT_MODE_VALUES((domestic || {}).mode).some(
			eachModePresent => isPresent(eachModePresent),
		);
 
		return isImportModePresent || isExportModePresent || isTransportModePresent;
	};
 
	/* search by route (import, export, transport) Europe, Middle East */
	isRoutePresent = (isPresent, preferences) => {
		const { import: importDetails, export: exportDetails } = preferences || {};
		const isImportRoutePresent = GET_SELECTED_ROUTE_VALUES((importDetails || {}).routes).some(
			eachModePresent => isPresent(eachModePresent),
		);
		const isExportRoutePresent = GET_SELECTED_ROUTE_VALUES((exportDetails || {}).routes).some(
			eachModePresent => isPresent(eachModePresent),
		);
		return isImportRoutePresent || isExportRoutePresent;
	};
 
	filterVendorTable = (searchValue, clientData) => {
		const filteredItems = (clientData || []).filter(eachItem => {
			if (searchValue === '') return true;
 
			const { company, users, preferences } = eachItem || {};
			const companyName = ((company || {}).name || '').toString();
 
			const isPresent = valuePassed => valuePassed
				.trim()
				.toLowerCase()
				.indexOf(searchValue.trim().toLowerCase()) !== -1;
 
			/* search by user's name, email */
			const isUserPresent = (users || []).some(
				({ name, email }) => isPresent(name) || isPresent(email),
			);
 
			if (
				isPresent(companyName)
				|| isUserPresent
				|| this.isServicePresent(isPresent, preferences)
				|| this.isModePresent(isPresent, preferences)
				|| this.isRoutePresent(isPresent, preferences)
			) {
				return true;
			}
			return false;
		});
 
		return filteredItems;
	};
 
	handleSearchChange = value => {
		const { vendorData } = this.props;
		this.setState({
			searchValue: value,
			filteredTableData: this.filterVendorTable(value, vendorData),
		});
	};
 
	render() {
		const { currentGroupId, isClient } = this.props;
		const { searchValue, filteredTableData, isUploadModalVisible } = this.state;
		const {
			isEditEnabled, columnsHeader, showVendorCreateDrawer, expandedRowRender,
		} = this.props;
 
		const menu = (
			<Menu>
				<Menu.Item id="vendor-bulk-upload-from-dropdown" onClick={this.showVendorUploadModal}>
					<i className="fas fa-cloud-upload-alt m-r-5" />
					Vendor Bulk Upload
				</Menu.Item>
			</Menu>
		);
 
		return (
			<>
				<Table
					columns={columnsHeader}
					dataSource={filteredTableData}
					rowKey="id"
					expandedRowRender={expandedRowRender}
					scroll={{ x: 300 }}
					pagination={{ pageSize: 100 }}
					className="sh-ant-table vendor-table-exapandable"
					title={() => (
						<Row type="flex">
							<Col lg={isEditEnabled ? 14 : 19} md={isEditEnabled ? 10 : 18} xs={24}>
								<div className="tab-header">Vendor List</div>
							</Col>
 
							<Col lg={5} md={6} xs={24}>
								<Search
									placeholder="Search"
									value={searchValue}
									onSearchChange={this.handleSearchChange}
								/>
							</Col>
 
							{isEditEnabled && (
								<Col lg={5} md={8} xs={24}>
									<CustomButtonGroup>
										<Button
											type="primary"
											id="group-create-vendor"
											onClick={showVendorCreateDrawer}
										>
											Create Vendor Company
										</Button>
 
										<Dropdown
											overlay={menu}
											trigger={['click']}
											overlayClassName="custom-dropdown"
											id="vendor-bulk-upload"
										>
											<Button type="primary">
												<Icon type="ellipsis" />
											</Button>
										</Dropdown>
									</CustomButtonGroup>
								</Col>
							)}
						</Row>
					)}
				/>
 
				{isUploadModalVisible && (
					<UploadExcel
						type="vendor-master"
						isVisible={isUploadModalVisible}
						hideUploadModal={this.hideUploadRateModal}
						extraParams={{ currentGroupId, userType: isClient ? 'client' : 'gocomet' }}
					/>
				)}
			</>
		);
	}
}
 
ListTable.propTypes = {
	isClient: PropTypes.bool,
	currentGroupId: PropTypes.string,
	isEditEnabled: PropTypes.bool.isRequired,
	vendorData: PropTypes.element,
	columnsHeader: PropTypes.element,
	showVendorCreateDrawer: PropTypes.func,
	expandedRowRender: PropTypes.element,
};
 
ListTable.defaultProps = {
	isClient: true,
	currentGroupId: '',
	vendorData: [],
	columnsHeader: [],
	expandedRowRender: {},
	showVendorCreateDrawer: () => {},
};
 
const mapStateToProps = state => {
	const { currentGroup: { id: currentGroupId = '' } = {} } = state.groups;
	return { currentGroupId };
};
 
export default connect(mapStateToProps)(ListTable);
