import { useState } from 'react';
import PropTypes from 'prop-types';
import { connect } from 'react-redux';
import { Button } from 'antd';
import { TextButton } from 'common-util/buttons';
import DataForm from 'common-util/form/DataForm';
import UsetifulBeaconComponent from 'common-util/AbTesting/UstifulBeaconComponent';
import { QuickAddWrapper } from './styles';
 
const QuickAddTracking = ({
	handleSubmit,
	showAddContainerModal,
	currentSelectedCarrier,
	groups,
	layout,
}) => {
	const [selectedCarrier, setSelectedCarrier] = useState(undefined);
	const [trackingNumber, setTrackingNumber] = useState(undefined);
	const [referenceNumber, setReferenceNumber] = useState('');
 
	const onCarrierSelection = value => {
		setSelectedCarrier(value);
	};
 
	const onTrackingNumberInput = value => {
		setTrackingNumber(value);
	};
 
	const onRefNumberInput = value => {
		setReferenceNumber(value);
	};
 
	const shouldShowPremiumCarriers = groups?.some(
		group => group.client_configuration?.container_tracking?.show_premium_carriers,
	);
 
	const quickAddContainerControls = [
		{
			type: 'input',
			inputType: 'text',
			name: 'containerNumber',
			placeholder: 'Tracking Number *',
			rules: [{ required: true, message: 'Container Number is required' }],
			initialValue: trackingNumber,
			onChange: e => onTrackingNumberInput(e.target.value),
		},
		{
			type: 'shipping-line-select',
			name: 'carrier_codes',
			multiple: false,
			placeholder: 'Select a Carrier *',
			rules: [{ required: true, message: 'Carrier is required' }],
			initialValue: selectedCarrier,
			carrierType: 'ocean',
			getPopupContainer: () => document.body,
			onChange: onCarrierSelection,
		},
		{
			type: 'input',
			inputType: 'input',
			name: 'referenceNumber',
			placeholder: 'Reference number',
			initialValue: referenceNumber,
			prefix: <UsetifulBeaconComponent />,
			onChange: e => onRefNumberInput(e.target.value),
		},
	];
 
	return (
		<QuickAddWrapper id="quick-add" layout={layout}>
			<p className="quick-add-heading">Quick Add</p>
 
			<div className="input-section-quick-add">
				<div className="form-section">
					<DataForm
						layout={layout === 'vertical' ? 'vertical' : 'inline'}
						controls={quickAddContainerControls}
						submitId="quick-tracking-add-MBL/Container-submit"
						onSubmit={handleSubmit}
						id="quick-tracking-add-MBL/Container-form"
						showSubmit={false}
					/>
					{layout === 'horizontal' && (
						<div className="quickadd-cta-wrapper">
							<div className="quickadd-text-cta">
								<div className="btn-container">
									<a
										className="see-all-carrier"
										href="/tools/payment?modal=all-carriers"
										rel="noopener noreferrer"
										target="_blank"
										id="quick-tracking-add-see-all-carrier-clicked"
									>
										See All Carriers <i className="fa fa-external-link-alt" />
									</a>
								</div>
								<div className="btn-container">
									<TextButton type="button" onClick={showAddContainerModal}>
										Add Additional Details &rarr;
									</TextButton>
								</div>
							</div>
						</div>
					)}
				</div>
				{currentSelectedCarrier?.premium_carrier && !shouldShowPremiumCarriers ? (
					<Button
						type="primary"
						className="add-button"
						href={`/tools/payment?recommended=business&selectedCarrier=${currentSelectedCarrier?.display_name
							|| currentSelectedCarrier.name
							|| 'Your carrier'}`}
						id="quick-tracking-add-upgrade-button-clicked"
					>
						<img src="/static/images/tracking/crown-icon.svg" alt="premium" /> Upgrade
					</Button>
				) : (
					<Button
						type="primary"
						className="add-button"
						htmlType="submit"
						form="quick-tracking-add-MBL/Container-form"
						id="quick-tracking-add-button-clicked"
					>
						Quick Add
					</Button>
				)}
			</div>
		</QuickAddWrapper>
	);
};
 
QuickAddTracking.propTypes = {
	layout: PropTypes.string,
	handleSubmit: PropTypes.func,
	showAddContainerModal: PropTypes.func,
	currentSelectedCarrier: PropTypes.shapeOf(PropTypes.object),
	groups: PropTypes.element,
};
 
QuickAddTracking.defaultProps = {
	layout: 'horizontal',
	handleSubmit: () => {},
	showAddContainerModal: () => {},
	currentSelectedCarrier: null,
	groups: [],
};
 
const mapStateToProps = state => {
	const currentSelectedCarrier = state.tracking?.addTrackingSelectedCarrier;
	const { groups } = state.users.profile_details || {};
	return {
		currentSelectedCarrier,
		groups,
	};
};
 
export default connect(
	mapStateToProps,
	{},
)(QuickAddTracking);
