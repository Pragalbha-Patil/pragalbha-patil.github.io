import { Component } from 'react';
import { connect } from 'react-redux';
import PropTypes from 'prop-types';
import { Tabs } from 'antd';
import { union } from 'lodash';
import {
	getVendors as getVendorsFn,
	sendEnquiryToVendors as sendEnquiryToVendorsFn,
} from 'store/quotes/actions';
import { getMeta } from 'util/redux-utils';
import { analyticsTrackEnquiry } from 'common-util/product/enquiry/functions';
import { VendorQuoteShimmer } from '../common/shimmer';
import AllVendors from './AllVendors';
import RfqSentTo from './RfqSentTo';
import { VendorQuotesDiv } from './styles';
import { EmptyMessage } from '../QuoteReceived/styles';
 
const { TabPane } = Tabs;
 
class VendorQuotes extends Component {
	handleRfqVendorSelection = {
		onSelect: (record, selected) => {
			const { enquiryType } = this.props;
			analyticsTrackEnquiry(
				`Enquiry All Vendor Checkbox Click - Single ${selected ? 'Select' : 'Unselect'}`,
				enquiryType,
			);
 
			const { selectedVendors = [] } = this.state;
			let newSelectedVendors = [];
			// If record is selected then add it's group_id to state else remove from state
			if (selected) {
				newSelectedVendors = [...selectedVendors, record.group_id];
			} else {
				newSelectedVendors = selectedVendors.filter(
					selectedVendor => selectedVendor !== record.group_id,
				);
			}
			this.setState({ selectedVendors: newSelectedVendors });
		},
		onSelectAll: (selected, selectedRows, changeRows) => {
			const { enquiryType } = this.props;
			analyticsTrackEnquiry(
				`Enquiry All Vendor Checkbox Click - All ${selected ? 'Select' : 'Unselect'}`,
				enquiryType,
			);
 
			const { selectedVendors } = this.state;
			let newSelectedVendors = [];
			// If all rows are selected then add all selectedRows in state else remove deselected rows
			if (selected) {
				const selectedVendorGroupIds = (selectedRows || []).map(selectedRow => selectedRow.group_id);
				newSelectedVendors = union(selectedVendors, selectedVendorGroupIds);
			} else {
				const deselectedVendorGroupIds = (changeRows || []).map(changeRow => changeRow.group_id);
				newSelectedVendors = selectedVendors.filter(
					selectedVendor => !deselectedVendorGroupIds.includes(selectedVendor),
				);
			}
			this.setState({ selectedVendors: newSelectedVendors });
		},
	};
 
	state = {
		selectedVendors: [],
	};
 
	websocket = 'undefined';
 
	componentDidMount() {
		const { getVendors, enquiryId } = this.props;
		getVendors(enquiryId);
	}
 
	handleToAllVendors = async () => {
		const { allVendorsList, sendEnquiryToVendors } = this.props;
		const totalVendorsId = allVendorsList.map(vendor => vendor.group_id);
		await sendEnquiryToVendors(totalVendorsId);
	};
 
	handleToSelectedVendors = async () => {
		const { selectedVendors } = this.state;
		const { sendEnquiryToVendors } = this.props;
		await sendEnquiryToVendors(selectedVendors);
	};
 
	handleRfqToSingleVendor = async vendorId => {
		const { sendEnquiryToVendors } = this.props;
		await sendEnquiryToVendors([vendorId]); /* sending to single vendor-id in array! */
	};
 
	render() {
		const { selectedVendors } = this.state;
		const {
			isLoading,
			isEditEnabled,
			enquiryType,
			freightNotRequiredMessage = '',
			allVendorsList = [],
			sentVendorsList = [],
		} = this.props;
 
		if (isLoading) {
			return <VendorQuoteShimmer isMessageLoading={isLoading} />;
		}
 
		return (
			<VendorQuotesDiv>
				{freightNotRequiredMessage && allVendorsList.length === 0 ? (
					<EmptyMessage>
						<h2>Freight Not Required</h2>
						<p>{freightNotRequiredMessage}</p>
					</EmptyMessage>
				) : (
					<Tabs
						defaultActiveKey={`${
							sentVendorsList.length > 0 ? 'tab-rfq-sent-to' : 'tab-all-vendors'
						}`}
					>
						<TabPane
							key="tab-all-vendors"
							tab={(
								<span
									data-analytics-id="enquiry-all-vendors-tab"
									data-analytics-select-type="parent"
									data-analytics-key-enquiry-type={enquiryType}
								>
									All Vendors&nbsp;({allVendorsList.length})
								</span>
							)}
						>
							<AllVendors
								isEditEnabled={isEditEnabled}
								allVendorsList={allVendorsList}
								selectedVendors={selectedVendors}
								handleToAllVendors={this.handleToAllVendors}
								handleToSelectedVendors={this.handleToSelectedVendors}
								handleRfqToSingleVendor={this.handleRfqToSingleVendor}
								handleRfqVendorSelection={this.handleRfqVendorSelection}
							/>
						</TabPane>
 
						<TabPane
							key="tab-rfq-sent-to"
							tab={(
								<span
									data-analytics-id="enquiry-rfq-sent-to-tab"
									data-analytics-select-type="parent"
									data-analytics-key-enquiry-type={enquiryType}
								>
									RFQ Sent to&nbsp;({sentVendorsList.length})
								</span>
							)}
						>
							<RfqSentTo isEditEnabled={isEditEnabled} sentVendorsList={sentVendorsList} />
						</TabPane>
					</Tabs>
				)}
			</VendorQuotesDiv>
		);
	}
}
 
VendorQuotes.propTypes = {
	isEditEnabled: PropTypes.bool.isRequired,
	isLoading: PropTypes.bool,
	enquiryId: PropTypes.string,
	enquiryType: PropTypes.string,
	allVendorsList: PropTypes.objectOf(Array),
	sentVendorsList: PropTypes.objectOf(Array),
	freightNotRequiredMessage: PropTypes.string,
	getVendors: PropTypes.func,
	sendEnquiryToVendors: PropTypes.func,
};
 
VendorQuotes.defaultProps = {
	isLoading: true,
	enquiryId: '',
	enquiryType: '',
	allVendorsList: [],
	sentVendorsList: [],
	freightNotRequiredMessage: '',
	getVendors: () => {},
	sendEnquiryToVendors: () => {},
};
 
const mapStateToProps = state => {
	const {
		allVendorsList, sentVendorsList, freightNotRequiredMessage, enquiryDetails,
	} =		state.quotes || {};
	const { enquiry_type } = enquiryDetails || {};
 
	return {
		allVendorsList,
		sentVendorsList,
		freightNotRequiredMessage,
		enquiryType: enquiry_type,
		...getMeta(state.quotes, 'getVendors'),
	};
};
 
export default connect(
	mapStateToProps,
	{
		getVendors: getVendorsFn,
		sendEnquiryToVendors: sendEnquiryToVendorsFn,
	},
)(VendorQuotes);
