import React, { Component } from 'react';
import propTypes from 'prop-types';
import { connect } from 'react-redux';
import {
	Row, Col, Input, Button, Card, Tag,
} from 'antd';
import {
	editSubCategoryName,
	addValueToArray,
	deleteSubcategory,
} from 'store/admin/name-mapper/actions';
import { keyToTitle } from 'components/client/invoice/common';
import { RedButton, ShadedButton } from 'common-util/buttons';
import { InputContainer } from '../../../styles';
 
class SubCategories extends Component {
	constructor(props) {
		super(props);
		const { data } = this.props;
		const { fixed_key } = data || {};
		this.state = {
			editing: false,
			inputHeading: fixed_key,
			matchingKeys: '',
			nonMatchingKeys: '',
		};
	}
 
	renderEdit = () => {
		this.setState({ editing: true });
	};
 
	cancelEdit = () => {
		this.setState({ editing: false });
	};
 
	saveAndHideEdit = () => {
		const { inputHeading } = this.state;
		const { editSubCategoryNameFn, data } = this.props;
		const { category, fixed_key } = data || {};
		const callBack = () => {
			editSubCategoryNameFn({ category, subCategory: fixed_key, newSubCategory: inputHeading });
		};
		this.setState(
			{
				editing: false,
			},
			callBack(),
		);
	};
 
	handleHeadingChange = value => {
		this.setState({ inputHeading: value });
	};
 
	saveMapping = (event, arrayName) => {
		const { value } = event.target || {};
		this.setState({
			[arrayName === 'matching_keys' ? 'matchingKeys' : 'nonMatchingKeys']: value,
		});
	};
 
	updateArrays = arrayName => {
		const { matchingKeys, nonMatchingKeys } = this.state;
		const { data, addValueToArrayFn } = this.props;
		const {
			category, fixed_key, matching_keys, non_matching_keys,
		} = data || {};
		const callBack = () => {
			addValueToArrayFn({
				category,
				subCategory: fixed_key,
				arrayName,
				values:
					arrayName === 'matching_keys'
						? [...matching_keys, matchingKeys]
						: [...non_matching_keys, nonMatchingKeys],
			});
		};
		this.setState(
			{
				[arrayName === 'matching_keys' ? 'matchingKeys' : 'nonMatchingKeys']: '',
			},
			callBack(),
		);
	};
 
	handleDelete = event => {
		event.stopPropagation();
		const { data, deleteSubcategoryFn } = this.props;
		const { category, fixed_key } = data || {};
		deleteSubcategoryFn({ category, subCategory: fixed_key });
	};
 
	handleTagClose = (key, arrayName) => {
		const { data, addValueToArrayFn } = this.props;
		const {
			category, fixed_key, matching_keys, non_matching_keys,
		} = data || {};
		let arrayToCopy =			arrayName === 'matching_keys' ? [...(matching_keys || [])] : [...(non_matching_keys || [])];
		arrayToCopy = arrayToCopy.filter(currentKey => currentKey !== key);
 
		const callBack = () => {
			addValueToArrayFn({
				category,
				subCategory: fixed_key,
				arrayName,
				values: [...arrayToCopy],
			});
		};
		this.setState(
			{
				[arrayName === 'matching_keys' ? 'matchingKeys' : 'nonMatchingKeys']: '',
			},
			callBack(),
		);
	};
 
	render() {
		const {
			editing, inputHeading, matchingKeys, nonMatchingKeys,
		} = this.state;
		const { data } = this.props;
		const { fixed_key, matching_keys, non_matching_keys } = data || {};
 
		return (
			<>
				<Card
					title={
						<>
							{!editing ? (
								<InputContainer>
									<p className="m-r-8">Sub - category: {keyToTitle(fixed_key)}</p>
									<ShadedButton onClick={this.renderEdit} size="small" className="m-r-8">
										<i className="fas fa-edit" />
										&nbsp;Edit
									</ShadedButton>
									<RedButton size="small" onClick={event => this.handleDelete(event)}>
										<i className="fas fa-trash" />
										&nbsp; Delete
									</RedButton>
								</InputContainer>
							) : (
								<InputContainer>
									<p className="m-r-8">Sub - category: </p>
									<Input
										className="m-r-8"
										value={inputHeading}
										onChange={event => this.handleHeadingChange((event.target || {}).value)}
									/>
									<Button type="primary" onClick={this.saveAndHideEdit} className="m-r-8">
										Save
									</Button>
									<ShadedButton onClick={this.cancelEdit}>Cancel</ShadedButton>
								</InputContainer>
							)}
						</>
					}
					size="small"
					className="m-b-8"
				>
					<Row>
						<Col xs={24} md={12}>
							<p>Matching keys:</p>
							{(matching_keys || []).map((key, index) => (
								<Tag
									closable
									key={`${key}-${index}`}
									className="m-t-8"
									onClose={() => this.handleTagClose(key, 'matching_keys')}
								>
									{keyToTitle(key || '')}
								</Tag>
							))}
							<InputContainer className="m-t-8">
								<Input
									placeholder="Add a key"
									className="m-r-8"
									onChange={event => this.saveMapping(event, 'matching_keys')}
									value={matchingKeys}
								/>
								<Button type="primary" onClick={() => this.updateArrays('matching_keys')}>
									Save
								</Button>
							</InputContainer>
						</Col>
						<Col xs={24} md={12}>
							<p>Non matching keys:</p>
							{(non_matching_keys || []).map((key, index) => (
								<Tag
									closable
									key={`${key}-${index}`}
									className="m-t-8"
									onClose={() => this.handleTagClose(key, 'non_matching_keys')}
								>
									{keyToTitle(key || '')}
								</Tag>
							))}
							<InputContainer className="m-t-8">
								<Input
									placeholder="Add a key"
									className="m-r-8"
									onChange={event => this.saveMapping(event, 'non_matching_keys')}
									value={nonMatchingKeys}
								/>
								<Button type="primary" onClick={() => this.updateArrays('non_matching_keys')}>
									Save
								</Button>
							</InputContainer>
						</Col>
					</Row>
				</Card>
			</>
		);
	}
}
 
SubCategories.propTypes = {
	data: propTypes.shape({}),
	editSubCategoryNameFn: propTypes.func,
	addValueToArrayFn: propTypes.func,
	deleteSubcategoryFn: propTypes.func,
};
 
SubCategories.defaultProps = {
	data: {},
	editSubCategoryNameFn: () => {},
	addValueToArrayFn: () => {},
	deleteSubcategoryFn: () => {},
};
 
export default connect(
	null,
	{
		editSubCategoryNameFn: editSubCategoryName,
		addValueToArrayFn: addValueToArray,
		deleteSubcategoryFn: deleteSubcategory,
	},
)(SubCategories);
