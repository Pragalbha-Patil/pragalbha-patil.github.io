import React, { Component } from 'react';
import PropTypes from 'prop-types';
import { connect } from 'react-redux';
import { getMeta } from 'util/redux-utils';
import {
	createConfiguration,
	editConfigurationDetails,
	updateCodeConfiguration,
	updateReportEditType,
} from 'store/admin/reports-dashboard/actions';
import Loader from 'common-util/loader';
import ModalHeader from './HeaderFooter/Header';
import ModalFooter from './HeaderFooter/Footer';
import PostConfiguration from './PostConfiguration';
import CodeView from './CodeView';
import AggregrationPipe from './AggregrationPipe';
import DataSheetTable from './DataSheetTable';
import UIView from './UIView';
import { isApiInitiating } from '../common';
import { CustomModal } from './styles';
 
class DetailsModal extends Component {
	state = {
		updatedConfiguration: null,
	};
 
	/* ------------------- helper GET functions ------------------- */
	getConfigurationValue = () => {
		const { updatedConfiguration } = this.state;
		const { codeConfiguration } = this.props;
		return updatedConfiguration !== null ? updatedConfiguration : codeConfiguration;
	};
 
	/* ------------------- event handling functions ------------------- */
	handleViewChange = value => {
		const { updateReportEditTypeFn } = this.props;
		updateReportEditTypeFn(value);
	};
 
	onConfigCodeUpdation = event => {
		event.persist();
		const { target } = event || {};
		const { value } = target || {};
		this.setState({ updatedConfiguration: value || '' });
	};
 
	/* ------------------- verify/edit/create API functions ------------------- */
	handleVerifyCode = () => {
		const { updateCodeConfigurationFn, editType } = this.props;
		const config = this.getConfigurationValue();
		if (editType === 'code' && config.length) {
			updateCodeConfigurationFn(config);
		}
	};
 
	handleSaveConfiguration = () => {
		const { reportId, editConfigurationDetailsFn, createConfigurationFn } = this.props;
 
		if (!reportId) {
			createConfigurationFn(this.getConfigurationValue());
		} else {
			editConfigurationDetailsFn(reportId);
		}
	};
 
	render() {
		const {
			requestStatus,
			messages,
			isVisible,
			editType,
			handleModalClose,
			codeConfiguration,
			aggregatePipelines,
			tables,
		} = this.props;
		const isLoading = isApiInitiating(requestStatus.getSheetsHistory);
		const isTableLoading = isApiInitiating(requestStatus.verifyConfigurationCode);
 
		let tableView = null;
		if ((tables || []).length !== 0) {
			tableView = (
				<DataSheetTable
					isTableLoading={isTableLoading}
					tables={tables}
					errorMessage={messages.verifyConfigurationCode || ''}
				/>
			);
		}
 
		return (
			<div>
				{isVisible && (
					<CustomModal
						visible={isVisible}
						onCancel={handleModalClose}
						title={<ModalHeader handleChange={this.handleViewChange} />}
						footer={[
							<ModalFooter
								key="report-dashboard-modal-footer"
								handleSaveConfiguration={this.handleSaveConfiguration}
							/>,
						]}
					>
						{isLoading ? (
							<div style={{ height: '100%' }}>
								<Loader />
							</div>
						) : (
							<>
								{editType === 'code' && (
									<>
										<CodeView
											codeConfiguration={codeConfiguration}
											handleVerifyCode={this.handleVerifyCode}
											onConfigCodeUpdation={this.onConfigCodeUpdation}
										/>
										{tableView}
									</>
								)}
								{editType === 'aggr_pipeline' && (
									<>
										<AggregrationPipe
											aggregatePipelines={aggregatePipelines}
											handleVerifyCode={this.handleVerifyCode}
										/>
										{tableView}
									</>
								)}
								{editType === 'ui' && <UIView />}
								{editType === 'post_config' && <PostConfiguration />}
							</>
						)}
					</CustomModal>
				)}
			</div>
		);
	}
}
 
DetailsModal.defaultProps = {
	requestStatus: {},
	messages: {},
	isVisible: false,
	editType: '',
	reportId: null,
	codeConfiguration: '',
	aggregatePipelines: '',
	tables: [],
	handleModalClose: () => {},
	createConfigurationFn: () => {},
	editConfigurationDetailsFn: () => {},
	updateCodeConfigurationFn: () => {},
	updateReportEditTypeFn: () => {},
};
 
DetailsModal.propTypes = {
	requestStatus: PropTypes.objectOf(PropTypes.shape({})),
	messages: PropTypes.objectOf(PropTypes.shape({})),
	isVisible: PropTypes.bool,
	editType: PropTypes.string,
	reportId: PropTypes.string,
	codeConfiguration: PropTypes.string,
	aggregatePipelines: PropTypes.string,
	tables: PropTypes.objectOf(PropTypes.array),
	handleModalClose: PropTypes.func,
	createConfigurationFn: PropTypes.func,
	editConfigurationDetailsFn: PropTypes.func,
	updateCodeConfigurationFn: PropTypes.func,
	updateReportEditTypeFn: PropTypes.func,
};
 
const mapStateToProps = state => {
	const {
		requestStatus,
		messages,
		reportId,
		editType,
		codeConfiguration,
		aggregatePipelines,
		tables,
	} = (state.admin || {}).reportsDashboard || {};
 
	return {
		requestStatus,
		messages,
		editType,
		reportId,
		codeConfiguration,
		aggregatePipelines,
		tables,
		...getMeta((state.admin || {}).reportsDashboard || {}, 'getSheetsHistory'),
	};
};
 
export default connect(
	mapStateToProps,
	{
		createConfigurationFn: createConfiguration,
		editConfigurationDetailsFn: editConfigurationDetails,
		updateCodeConfigurationFn: updateCodeConfiguration,
		updateReportEditTypeFn: updateReportEditType,
	},
)(DetailsModal);
