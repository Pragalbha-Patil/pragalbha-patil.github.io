import PropTypes from 'prop-types';
import { connect } from 'react-redux';
import Link from 'next/link';
import {
	Form, Input, Button, Icon,
} from 'antd';
import { forgotPassword as forgotPasswordFn } from 'store/users/actions';
import { LoginPage, LoginContainer, LoginBox } from 'components/public/login/styles';
 
const ForgotPassword = ({ form: { getFieldDecorator, validateFields }, forgotPassword }) => {
	const handleSubmit = e => {
		e.preventDefault();
		validateFields((err, values) => {
			if (!err) forgotPassword(values);
		});
	};
 
	return (
		<LoginPage>
			<LoginContainer>
				<LoginBox>
					<div>
						<h1>Forgot Password</h1> {/* Only for SEO | hidden in UI */}
						<div className="logo">
							<a href="/">
								<img src="/static/images/logo-dark.png" alt="company logo" />
							</a>
						</div>
						<div>
							<h2>Forgot your password?</h2>
							<h3>Enter your email below to receive a link to reset your password</h3>
						</div>
						<div className="form-container">
							<Form onSubmit={handleSubmit}>
								<Form.Item label="Your Email" required>
									{getFieldDecorator('email', {
										rules: [
											{
												required: true,
												message: 'Please enter your email',
											},
										],
									})(
										<Input
											type="email"
											size="large"
											prefix={<Icon type="user" style={{ color: 'rgba(0,0,0,.25)' }} />}
											placeholder="Enter your company email"
										/>,
									)}
								</Form.Item>
								<Form.Item>
									<Button type="primary" htmlType="submit" id="login-submit" size="large" block>
										Receive Link
									</Button>
								</Form.Item>
							</Form>
						</div>
					</div>
				</LoginBox>
				<p className="footer">
					Already know your password? Please&nbsp;
					<Link href="/login">
						<a>Login</a>
					</Link>
				</p>
			</LoginContainer>
		</LoginPage>
	);
};
 
ForgotPassword.propTypes = {
	form: PropTypes.shape({}).isRequired,
	forgotPassword: PropTypes.func.isRequired,
};
 
export default connect(
	undefined,
	{ forgotPassword: forgotPasswordFn },
)(Form.create({ name: 'forgot_password' })(ForgotPassword));
