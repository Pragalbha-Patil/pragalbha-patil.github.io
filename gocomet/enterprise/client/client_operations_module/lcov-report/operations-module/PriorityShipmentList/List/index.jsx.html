import PropTypes from 'prop-types';
import { Row, Tooltip } from 'antd';
import { connect } from 'react-redux';
import { LocationAddress } from 'common-util/reusable-components';
import { isDomesticSurfaceEnquiry } from 'common-util/product/enquiry/functions';
import EnquiriesLoader from 'common-util/shimmer/Enquiries';
import { openURL } from 'common-util/reusable-functions';
import { getDetails as getShipmentDetailsFn } from 'store/operations-module/actions';
import {
	RowTwo,
	CardColumn,
	QuoteName,
	CreatedBy,
	ShipmentInfo,
	NoData,
} from 'common-util/reusable-styles';
import {
	ListingContainer, ShipmentCard, EstArrival, StatusInfo,
} from '../styles';
 
const Shipments = ({
	isLoading, isListingPage, list, hideDrawer, getShipmentDetails,
}) => {
	const cardClick = key => {
		if (isListingPage) {
			openURL(`/operations-module/${key}`);
		} else {
			getShipmentDetails(key);
			hideDrawer();
		}
	};
 
	if (isLoading) return <EnquiriesLoader count={4} />;
 
	return (
		<ListingContainer isListingPage={isListingPage}>
			{(list || []).length === 0 ? (
				<NoData>No data found!</NoData>
			) : (
				<>
					{list.map(({
						id, key, name, est_arrival, consignee, status_details, enquiry,
					}) => {
						const { incoterm, mode, shipment_type } = enquiry || {};
 
						return (
							<ShipmentCard key={`${id}`} onClick={() => cardClick(key)}>
								<Row className="row-1" type="flex">
									<CardColumn xl={6} lg={6} md={6} xs={12}>
										<QuoteName>
											<Tooltip title={name || 'N/A'}>{name || 'N/A'}</Tooltip>
										</QuoteName>
									</CardColumn>
 
									<CardColumn xl={13} lg={13} md={6} xs={12}>
										<CreatedBy>
											<Tooltip title={(consignee || {}).name || 'N/A'}>
												Consignee:&nbsp;{(consignee || {}).name || 'N/A'}
											</Tooltip>
										</CreatedBy>
									</CardColumn>
 
									<EstArrival xl={5} lg={5} md={6} xs={12}>
										Est Arrival: {est_arrival || 'N/A'}
									</EstArrival>
								</Row>
 
								<RowTwo type="flex" className="row-2">
									<CardColumn xl={6} lg={6} md={12} sm={12} xs={24} className="no-pad-r">
										<ShipmentInfo mode={mode}>
											<span className="icons shipment" />
											<div>
												<p>
													{shipment_type} {!isDomesticSurfaceEnquiry(enquiry) && <>({incoterm})</>}{' '}
													{mode}
												</p>
												<p>GID: {key}</p>
											</div>
										</ShipmentInfo>
									</CardColumn>
 
									<CardColumn xl={10} lg={10} md={12} sm={12} xs={24}>
										<LocationAddress details={enquiry} />
									</CardColumn>
 
									<CardColumn xl={8} lg={8} md={12} xs={12} className="no-pad-r">
										<StatusInfo>
											<div>{(status_details || {}).status || 'N/A'}</div>
											<div>{(status_details || {}).info || 'N/A'}</div>
										</StatusInfo>
									</CardColumn>
								</RowTwo>
							</ShipmentCard>
						);
					})}
				</>
			)}
		</ListingContainer>
	);
};
 
Shipments.propTypes = {
	isLoading: PropTypes.bool.isRequired,
	isListingPage: PropTypes.bool,
	list: PropTypes.shape([]).isRequired,
	hideDrawer: PropTypes.func,
	getShipmentDetails: PropTypes.func,
};
 
Shipments.defaultProps = {
	isListingPage: false,
	hideDrawer: () => {},
	getShipmentDetails: () => {},
};
 
const mapStateToProps = state => {
	const { priority_shipments } = state.operationsModule || {};
	return { list: priority_shipments };
};
 
export default connect(
	mapStateToProps,
	{
		getShipmentDetails: getShipmentDetailsFn,
	},
)(Shipments);
