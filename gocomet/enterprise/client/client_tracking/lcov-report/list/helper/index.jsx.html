export const CONTAINER_NUMBER_REGEX = /^[a-z]{4}\d{7}$/;

export const BL_NEEDED_KEY = 'with_bl';
export const CONTAINER_AND_VESSEL_NEEDED_KEY = 'with_container_number_and_vessel';
export const VESSEL_NEEDED_KEY = 'with_vessel';

export const CARRIER_RATING_P3 = 'p3';
 
export const isContainerNumber = tracking_number => (
	CONTAINER_NUMBER_REGEX.test(tracking_number?.toLowerCase())
);
 
export const GET_TRACKING_NUMBER_LABEL = (mode, selectedCarrier, trackingType = 'pro') => {
	const { tracking_available, support_rating } = selectedCarrier || {};
	if (tracking_available === BL_TRACKING_NUMBER_NEEDED_KEY || tracking_available === BL_NEEDED_KEY) {
		return 'MBL/Booking Number';
	}
	if (!isAir(mode)) {
		if (isRoad(mode)) {
			return (`${(trackingType || '').toUpperCase()}`);
		}
		if (!isCourier(mode)) {
			return `${(support_rating !== CARRIER_RATING_P3) ? 'MBL/' : ''}Container Number`;
		}
		return 'Tracking Number';
	}
	return 'MAWB Number';
};
