import PropTypes from 'prop-types';
import { connect } from 'react-redux';
import { Button } from 'antd';
import CommentsAndPriceIncrease from './CommentsAndPriceIncrease';
import { IS_SUBMIT_DISABLED } from '../helper';
import { Footer } from './styles';
 
const TransitDetailsForm = ({
	isLoadingSubmit,
	errors,
	quote_form,
	enquiryDetails,
	closeModal,
	updateQuoteForm,
	handleSubmitQuote,
}) => {
	const { enquiry_type } = enquiryDetails || {};
	const {
		form_requisites, quote, require_price_increase_remarks,
	} =		quote_form || {};
	const {
		comments,
		price_increase_remarks,
		reason_for_price_increase,
	} = quote || {};
 
	const {
		reasons_for_price_increase: reason_price,
	} = form_requisites || {};
 
	return (
		<>
 
			<CommentsAndPriceIncrease
				comments={comments}
				isPriceRemarksRequired={require_price_increase_remarks}
				hasAnyFormError={(errors || {}).any}
				price_increase_remarks={price_increase_remarks}
				reason_for_price_increase={reason_for_price_increase}
				reason_price={reason_price}
				updateQuoteForm={updateQuoteForm}
			/>
 
			<Footer>
 
				<Button
					type="primary"
					loading={isLoadingSubmit}
					onClick={handleSubmitQuote}
					disabled={IS_SUBMIT_DISABLED(isLoadingSubmit, enquiryDetails, quote)}
					data-analytics-id="enquiry-submit-quote"
					data-analytics-key-enquiry-type={enquiry_type}
					data-analytics-key-side-type="left"
				>
					Save & Submit Quote
				</Button>
 
				<Button onClick={closeModal} type="ghost">
					Close
				</Button>
			</Footer>
		</>
	);
};
 
TransitDetailsForm.propTypes = {
	isLoadingSubmit: PropTypes.bool,
	errors: PropTypes.shape({}),
	quote_form: PropTypes.shape({}),
	enquiryDetails: PropTypes.shape({}),
	closeModal: PropTypes.func,
	updateQuoteForm: PropTypes.func,
	handleSubmitQuote: PropTypes.func,
};
 
TransitDetailsForm.defaultProps = {
	isLoadingSubmit: false,
	errors: {},
	quote_form: {},
	enquiryDetails: {},
	closeModal: () => {},
	updateQuoteForm: () => {},
	handleSubmitQuote: () => {},
};
 
const mapStateToProps = state => {
	const { enquiryDetails, quote_form } = (state || {}).quotes || {};
	return {
		quote_form,
		enquiryDetails,
	};
};
 
export default connect(
	mapStateToProps,
	{},
)(TransitDetailsForm);
