const getLaggingShiploadNumber = (shiploads, laggingShiploadContainerNo) => {
	const laggingShiploadNumber = shiploads.findIndex(
		shipload => shipload.container_number === laggingShiploadContainerNo,
	);
	return laggingShiploadNumber + 1;
};
 
const currentDemurrageData = (
	events,
	insightDemurrageDays,
	insightDemurrageCost,
	shiploads,
	laggingShiploadContainerNo,
) => {
	const demurrageDays = Math.floor(insightDemurrageDays);
	const isGateOutOrDelivered = isGateOut(events) || isDelivered(events);
	const demurrageText = isGateOutOrDelivered ? 'Demurrage' : 'Probable Demurrage';
	const laggingShiploadNo = getLaggingShiploadNumber(shiploads, laggingShiploadContainerNo);
	let demurrageInsightText = '';
	if (demurrageDays < 0) {
		if (shiploads.length > 1) {
			demurrageInsightText = (
				<span>
					Shipload {laggingShiploadNo} (Free days Left):{' '}
					<b>
						{' '}
						&lt; {demurrageDays * -1} {demurrageDays * -1 === 1 ? 'day' : 'days'}...
					</b>
				</span>
			);
		} else {
			demurrageInsightText = (
				<span>
					Free days Left:{' '}
					<b>
						{' '}
						&lt; {demurrageDays * -1} {demurrageDays * -1 === 1 ? 'day' : 'days'}
					</b>
				</span>
			);
		}
	} else if (shiploads.length > 1) {
		demurrageInsightText = (
			<span>
				Shipload {laggingShiploadNo} ({demurrageText}):{' '}
				<b>
					{' '}
					&gt; {demurrageDays} {demurrageDays <= 1 ? 'day' : 'days'}...
				</b>
			</span>
		);
	} else {
		demurrageInsightText = (
			<span>
				{demurrageText}:{' '}
				<b>
					{' '}
					&gt; {demurrageDays} {demurrageDays <= 1 ? 'day' : 'days'}
				</b>
			</span>
		);
	}
 
	if (shiploads.length <= 1) {
		return (
			<InsightText color="initial">
				<span>&nbsp;{demurrageInsightText}</span>
				{insightDemurrageCost !== null && (
					<span className="m-l-8 demurrage-bar">
						&nbsp;&nbsp;&nbsp;&nbsp;Cost Incurred:&nbsp;&nbsp;<b>${insightDemurrageCost}</b>
					</span>
				)}
			</InsightText>
		);
	}
 
	let totalInsightDemurrageCost = 0;
	const shiploadsDD = shiploads.map((shipload, i) => {
		const { demurrage_days } = shipload?.stats?.detention_and_demurrage || {};
		const shiploadDemurrageDays = Math.floor(demurrage_days);
		const isShiploadGateOutOrDelivered = isGateOut(shipload.events) || isDelivered(shipload.events);
		const shiploadDemurrageText = isShiploadGateOutOrDelivered ? 'Demurrage' : 'Probable Demurrage';
		let demurrageDaysText = '';
		if (shiploadDemurrageDays < 0) {
			demurrageDaysText = (
				<span>
					{' '}
					Shipload {i + 1} (Free Days Left):
					<b>
						{' '}
						&lt; {shiploadDemurrageDays * -1} {shiploadDemurrageDays * -1 === 1 ? 'day' : 'days'}{' '}
					</b>
				</span>
			);
		} else {
			demurrageDaysText = (
				<span>
					{' '}
					Shipload {i + 1} ({shiploadDemurrageText}):
					<b>
						{' '}
						&gt; {shiploadDemurrageDays} {shiploadDemurrageDays <= 1 ? 'day' : 'days'}
					</b>
				</span>
			);
		}
		const { demurrage_cost_incurred } = shipload?.stats?.detention_and_demurrage || {};
		const demurrageCostText = (
			<span>
				{' '}
				Shipload {i + 1}:<b> ${demurrage_cost_incurred} </b>{' '}
			</span>
		);
		totalInsightDemurrageCost += demurrage_cost_incurred || 0;
		return { demurrageDaysText, demurrageCostText };
	});
 
	return (
		<>
			<InsightText color="initial">
				<>
					<DemurrageDays demurrageInsightText={demurrageInsightText} shiploadsDD={shiploadsDD} />
					{insightDemurrageCost !== null && (
						<DemurrageCost
							totalInsightDemurrageCost={totalInsightDemurrageCost}
							shiploadsDD={shiploadsDD}
						/>
					)}
				</>
			</InsightText>
		</>
	);
};
const insightInvalidTooltip = (showInvalidTooltip, remark, showValidTillDate, tag) => {
	if (remark && (tag || '').toLowerCase() === 'invalid') {
		return (
			<InvalidText>
				<Tooltip
					placement="bottom"
					title={
						showInvalidTooltip && (
							<span>
								GoComet will actively track this shipment till <b>{showValidTillDate}</b>
							</span>
						)
					}
				>
					<img
						src="/static/images/tracking/Invalid-No_ship_Icon.svg"
						alt="Invalid ship"
						width="50"
						height="50"
					/>
					{remark}.
					<a
						target="_blank"
						href="https://gocomet.freshdesk.com/support/solutions/articles/43000664638#Your%20entered%20POL%20and%20POD%20not[â€¦]hed%20with%20carrier%20POL%20and%20POD"
						rel="noopener noreferrer"
					>
						{' '}
						&nbsp;Refer to this doc for more understanding &nbsp;
						<i className="fas fa-external-link-alt" />
					</a>
				</Tooltip>
			</InvalidText>
		);
	}
	if (remark) {
		return <InsightText color={INSIGHT_TEXT_COLOR[tag?.toLowerCase()]}>{remark}</InsightText>;
	}
	return null;
};

const insight = (
	events,
	isTrackingTrialEnabled,
	tag = '',
	remark,
	insightDemurrageDays,
	insightDemurrageCost,
	shiploads,
	laggingShiploadContainerNo,
	showInvalidTooltip,
	showValidTillDate,
) => (
	<InsightContainer>
		<InsightContent>{GET_INSIGHTBAR(tag)}</InsightContent>
		<InsightContent
			blurred={isTrackingTrialEnabled}
			className={insightDemurrageCost !== 0 ? 'full-width' : 'no-style-width'}
		>
			{isTrackingTrialEnabled && tag?.toLowerCase() === 'geofence' ? (
				<>
					<div>
						Latest Insight : &nbsp;
						<InsightText color={INSIGHT_TEXT_COLOR[tag?.toLowerCase()]}>{remark}</InsightText>
					</div>
					<BlurredBox>
						<span>
							<i className="fa fa-unlock-alt unlock-icon" aria-hidden="true" />
							Unlock Geofence insights on your vessel&apos;s live location
						</span>
						<ContactUsBtn
							Component={props => (
								<Button type="primary" id="geofence-learn-more-trial-user-summary-click" {...props}>
									Learn more
								</Button>
							)}
							pageName="Geofence for EGA trial user"
						/>
					</BlurredBox>
				</>
			) : (
				<>
					{tag === 'demurrage' ? (
						<InsightText color="initial">
							{currentDemurrageData(
								events,
								insightDemurrageDays,
								insightDemurrageCost,
								shiploads,
								laggingShiploadContainerNo,
							)}
						</InsightText>
					) : (
						insightInvalidTooltip(showInvalidTooltip, remark, showValidTillDate, tag)
					)}
				</>
			)}
		</InsightContent>
	</InsightContainer>
);
