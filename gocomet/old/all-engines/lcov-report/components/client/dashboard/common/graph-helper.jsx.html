/* global am4core am4charts */
import { COLOR } from 'util/theme';
 
export const checkGraph = chartId => typeof am4core === 'undefined'
	|| typeof am4charts === 'undefined'
	|| typeof document === 'undefined'
	|| !document.getElementById(chartId);
 
/**
 * https://www.amcharts.com/demos/stacked-column-chart/
 */
export const renderStackedBar = ({
	chartId,
	graphData,
	category,
	categoryName,
	xText,
	yText,
	stackedSeries,
	isPercentage = false,
}) => {
	if (checkGraph(chartId)) {
		return null;
	}
 
	const chart = am4core.create(chartId, am4charts.XYChart);
	chart.data = graphData;
 
	/* Create axes */
	const categoryAxis = chart.xAxes.push(new am4charts.CategoryAxis());
	categoryAxis.dataFields.category = category;
	categoryAxis.dataFields.categoryName =  categoryName || null;
	categoryAxis.title.text = xText || null;
	if(categoryName)
	categoryAxis.tooltip.disabled = true;
	
	/* truncating labels */
	categoryAxis.renderer.minGridDistance = 20; /* necessary for truncating text */
	const label = categoryAxis.renderer.labels.template;
	label.truncate = true;
	label.maxWidth = 90;
	label.tooltipText = categoryName ? '{categoryName}' : '{category}';
 
	const valueAxis = chart.yAxes.push(new am4charts.ValueAxis());
	valueAxis.min = 0;
	if (isPercentage) {
		valueAxis.max = 100;
	}
	valueAxis.title.text = yText || null;
 
	/* Create series */
	function createSeries(field, name) {
		const series = chart.series.push(new am4charts.ColumnSeries());
		series.name = name;
		series.dataFields.valueY = field;
		series.dataFields.categoryX = category;
		series.dataFields.categoryName = categoryName;
		series.sequencedInterpolation = true;
		series.stacked = true;
		series.columns.template.width = am4core.percent(60);
		//assumed that there are only 2 values for the field in the tracking
		//percentage_on_time_shipments and percentage_delayed_shipments
		//out of which percentage_on_time_shipments is assumed to be greater than 50%
		//which makes the tooltip UI look nice.
		if(field === 'percentage_on_time_shipments')
		series.columns.template.tooltipText = '{categoryName}';
 
		const labelBullet = series.bullets.push(new am4charts.LabelBullet());
		labelBullet.label.text = isPercentage ? '{valueY}%' : '{valueY}';
		labelBullet.locationY = 0.5;
		labelBullet.label.hideOversized = true; /* hides label eg. 0, 1, 2 */
 
		return series;
	}
 
	(stackedSeries || []).forEach(({ key, text }) => {
		createSeries(key, text);
	});
 
	chart.legend = new am4charts.Legend();
 
	chart.cursor = new am4charts.XYCursor();
	chart.cursor.fullWidthLineX = true;
	chart.cursor.lineX.strokeOpacity = 0;
	chart.cursor.lineX.fill = am4core.color(COLOR.BLACK);
	chart.cursor.lineX.fillOpacity = 0.1;
	chart.cursor.lineY.disabled = true;
 
	valueAxis.cursorTooltipEnabled = false;
 
	return chart;
};
